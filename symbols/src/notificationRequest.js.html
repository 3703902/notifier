1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the NotficationRequest prototype and its methods.
  5  */
  6 module.exports = NotificationRequest;
  7 
  8 /**
  9  * NotificationRequest constructor
 10  *
 11  * @class
 12  *
 13  * A NotificationRequest represents a notification request.
 14  *
 15  * @param {String} userId user to notify
 16  * @param {String} token the value used to reach a specific device
 17  * @param {Object} content the notification payload to send to the device. The 'content' attribute holds the actual
 18  *  payload to deliver. Additional attributes may exist depending on the os-specific notification service being used
 19  *  for delivery.
 20  * @param {Number} secondsToLive how long this notification is valid
 21  * @param {Function} removeRegistrationProc function to call to remove a token from further notifications
 22  */
 23 function NotificationRequest(userId, token, content, secondsToLive, removeRegistrationProc) {
 24     this.userId = userId;
 25     this.token = token;
 26     this.content = content;
 27     this.secondsToLive = secondsToLive;
 28     this.removeRegistrationProc = removeRegistrationProc;
 29 }
