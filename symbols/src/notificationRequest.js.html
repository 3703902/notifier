1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the NotficationRequest prototype and its methods.
  5  */
  6 module.exports = NotificationRequest;
  7 
  8 /**
  9  * NotificationRequest constructor
 10  *
 11  * @class
 12  *
 13  * A NotificationRequest represents a notification request from an extenal entity.
 14  *
 15  * @param {String} registrationId the registration of a user that was used to select a payload
 16  * @param {String} service the OS-specific service being used for notification delivery
 17  * @param {String} token the OS-specifc identifier used for notification delivery to a specific device
 18  * @param {Object} template defines the notification payload to generate.
 19  */
 20 function NotificationRequest(registrationId, service, token, template) {
 21     this.registrationId = registrationId;
 22     this.service = service;
 23     this.token = token;
 24     this.template = template;
 25     this.content = null;
 26 }
 27 
 28 NotificationRequest.prototype = {
 29 
 30     /**
 31      * Prepare a notification payload using the given values. Updates the content attribute with the payload text.
 32      *
 33      * @param {Object} substitutions mapping of placeholder names and the values to use when substituting placeholder
 34      * tags within a notification template.
 35      *
 36      * @param {String} userId the ID of the user being notified
 37      * @param {RegistrationStore} registrationStore the storage used for registrations. Used if the OS-specific service
 38      * flags a token as invalid
 39      * @param {Integer} secondsToLive number of seconds this request is valid
 40      */
 41     prepare: function(substitutions, userId, registrationStore, secondsToLive) {
 42         this.userId = userId;
 43         this.registrationStore = registrationStore;
 44         this.secondsToLive = secondsToLive;
 45         // Generate the notification payload from the parsed template and the given substitution values.
 46         this.content = this.template.content(substitutions);
 47     },
 48 
 49     /**
 50      * Remove the route used to deliver the last notification as it was deemed invalid by the OS-specific service.
 51      */
 52     forgetRoute: function () {
 53         this.registrationStore.delRoute(this.userId, this.registrationId, this.token);
 54     }
 55 };
