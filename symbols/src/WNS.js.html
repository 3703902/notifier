1 /**
  2  * @fileOverview Defines the WNS prototype and its methods.
  3  */
  4 module.exports = WNS;
  5 
  6 var config = require('./config');
  7 var fs = require('fs');
  8 var HTTPRequest = require('./httpRequest');
  9 var Q = require('q');
 10 
 11 /**
 12  * WNS constructor.
 13  *
 14  * @class
 15  *
 16  * A WNS object sends XML payloads to a WNS server for delivery to a specific Windows 8 device.
 17  */
 18 function WNS() {
 19     this.log = config.log('WNS');
 20     this.accessToken = null;
 21     this.accessTokenLifetime = null;
 22     this.pending = null;
 23     fs.readFile(config.wns_access_token_cache, function (err, data) {
 24         if (data) {
 25             this.log.info('loading access token from', config.wns_access_token_cache);
 26             try {
 27                 data = JSON.parse(data);
 28                 this.accessToken = data.accessToken;
 29                 this.accessTokenLifetime = data.accessTokenLifetime;
 30                 this.log.info('accessToken:', this.accessToken);
 31                 this.log.info('accessTokenLifetime:', this.accessTokenLifetime);
 32             } catch (x) {
 33                 this.log.error('failed loading from', config.wns_access_token_cache);
 34             }
 35         }
 36     }.bind(this));
 37 }
 38 
 39 /**
 40  * WNS prototype.
 41  *
 42  * Defines the methods available to a WNS instance.
 43  */
 44 WNS.prototype = {
 45 
 46     fetchAccessToken: function(params) {
 47         return HTTPRequest(params,
 48                            function(err, resp, body) {
 49                                body = JSON.parse(body);
 50                                return body.access_token;
 51                            });
 52     },
 53 
 54     /**
 55      * Send a notification to a client via WNS.
 56      *
 57      * @param {NotificationRequest} req the identifier of the client to notify
 58      */
 59     sendNotification: function(req) {
 60         var log = this.log.child('sendNotification');
 61         var self = this;
 62         log.BEGIN(req);
 63 
 64         if (this.pending !== null) {
 65             // *** We should have a pending accessToken refresh in-progress. If not, then we are sunk.
 66             this.pending.push(req);
 67             return;
 68         }
 69 
 70         if (this.accessToken === null || this.accessTokenLifetime < Date.now()) {
 71             log.info('renewing access token:', this.accessToken);
 72             this.pending = [];  // *** This keeps others from entering this branch until the renewal is done
 73             this.pending.push(req);
 74             this.fetchAccessToken({
 75                                       'secondsToLive': 0,
 76                                       'retryTimeout': 500,
 77                                       'retryStatusCheck': function (statusCode) { return statusCode !== 200; },
 78                                       'uri': config.wns_access_token_url,
 79                                       'method': 'POST',
 80                                       'form': {
 81                                           'grant_type': 'client_credentials',
 82                                           'scope': 'notify.windows.com',
 83                                           'client_id': config.wns_package_sid,
 84                                           'client_secret': config.wns_client_secret
 85                                       }
 86                                   })
 87                 .then(function (accessToken) {
 88                           log.info('new accessToken:', accessToken);
 89                           self.accessToken = accessToken;
 90                           self.accessTokenLifetime = Date.now() + 24 * 60 * 60 * 1000;
 91 
 92                           // Save to local cache in case we restart
 93                           fs.writeFile(config.wns_access_token_cache,
 94                                        JSON.stringify(
 95                                        {
 96                                            'accessToken': self.accessToken,
 97                                            'accessTokenLifetime': self.accessTokenLifetime
 98                                        }),
 99                                       function (err) {
100                                           log.info('wrote accessToken to', config.wns_access_token_cache, err);
101                                       });
102 
103                           // If we have pending notifications waiting on a valid access token, resubmit them now.
104                           var pending = self.pending;
105                           self.pending = null;
106                           pending.forEach(function (item) {
107                                               self.sendNotification(item);
108                                           });
109                           return accessToken;
110                       })
111                 .end();
112             return;
113         }
114 
115         // Post the notification now that we have a valid access token
116         HTTPRequest({
117                         'secondsToLive': 30,
118                         'retryTimeout': 500,
119                         'uri': req.token,
120                         'method': 'POST',
121                         'body': req.content.text,
122                         'forever': true, // enable long-lived connections
123                         'headers': {
124                             'Content-Type': 'text/xml',
125                             'X-WNS-Type': req.content.kind,
126                             'X-WNS-RequestForStatus': 'true',
127                             'Authorization': 'Bearer ' + this.accessToken,
128                             'Host': 'cloud.notify.windows.com'
129                         }
130                     },
131                     function(err, resp, body) {
132                         log.debug(err, resp, body);
133                         if (err !== null) {
134                             log.error('failed to deliver notification:', err);
135                             return err;
136                         }
137                         else {
138                             log.debug('response code:', resp.statusCode);
139                             switch (resp.statusCode) {
140                             case 200:
141                                 log.info('delivered notification');
142                                 break;
143                             case 401: // Unauthorized
144                             case 403: // Forbidden
145                                 // This *really* should not happen here.
146                                 log.error('access token has strangely expired');
147                                 self.accessToken = null;
148                                 return self.sendNotification(req);
149 
150                             case 410: // Gone
151                                 log.warn('user registration is no longer valid');
152                                 req.removeRegistrationProc();
153                                 break;
154 
155                             default:
156                                 log.warn('failed to deliver notification:', resp.statusCode);
157                             }
158                             return resp;
159                         }
160                     }).end();
161     }
162 };
