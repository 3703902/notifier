1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the WNS prototype and its methods.
  5  */
  6 module.exports = WNS;
  7 
  8 var config = require('./config');
  9 var wns = require('wns');
 10 
 11 /**
 12  * WNS constructor.
 13  *
 14  * @class
 15  *
 16  * A WNS object sends XML payloads to a WNS server for delivery to a specific Windows 8 device.
 17  */
 18 function WNS(monitorManager) {
 19     this.log = config.log('WNS');
 20     this.monitorManager = monitorManager;
 21 }
 22 
 23 /**
 24  * WNS prototype.
 25  *
 26  * Defines the methods available to a WNS instance.
 27  */
 28 WNS.prototype = {
 29 
 30     /**
 31      * Send a notification to a client via WNS.
 32      *
 33      * @param {NotificationRequest} req the identifier of the client to notify
 34      */
 35     sendNotification: function(req) {
 36         try {
 37 
 38             var log = this.log.child('sendNotification');
 39             var self = this;
 40             log.BEGIN();
 41 
 42             log.info('req.token:', req.token);
 43             log.info('req.content:', req.payload);
 44             log.info('req.template.kind:', req.template.kind);
 45 
 46             var options = {
 47                 client_id: config.wns_package_sid,
 48                 client_secret: config.wns_client_secret,
 49                 headers: {'X-WNS-RequestForStatus': 'true'}
 50             };
 51 
 52             if (req.template.kind == 'wns/raw') {
 53                 options.headers['Content-Type'] = 'application/octet-stream';
 54             }
 55             else {
 56                 options.headers['Content-Type'] = 'text/xml';
 57             }
 58 
 59             log.info('options:', options);
 60 
 61             wns.send(req.token, req.payload, req.template.kind, options, function(error, result) {
 62                 try {
 63                     log.info('error:', error);
 64                     log.info('result:', result);
 65                 }
 66                 catch (x) {
 67                     log.error(x);
 68                 }
 69             });
 70         }
 71         catch (x) {
 72             log.error(x);
 73         }
 74     }
 75 };
