1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the PostTracker prototype and its methods.
  5  */
  6 module.exports = PostTracker;
  7 
  8 var Deque = require('./deque');
  9 var config = require('./config');
 10 
 11 /**
 12  * PostTracker constructor
 13  *
 14  * @class
 15  *
 16  * A PostTracker instance maintains a LIFO size-limited queue that contains Node instances that describe when a
 17  * notification request was received for processing. New Node instances are added using the 'add' method. The 'track'
 18  * method looks for cached Node instances and if found logs the duration in seconds between the receipt time of the
 19  * notification and the initiation time stored in the Node instance.
 20  */
 21 function PostTracker(maxSize) {
 22     this.log = config.log('PostTracker');
 23     this.maxSize = maxSize;
 24     this.active = new Deque();
 25     this.mapping = {};
 26 }
 27 
 28 /**
 29  * PostTracker.Node constructor
 30  *
 31  * @class
 32  *
 33  * A PostTracker.Node instance contains attributes associated with a post notification request. In particular, it
 34  * tracks the unique sequence ID for the request, and the time when the request was made.
 35  */
 36 PostTracker.Node = function(sequenceId, when) {
 37     this.sequence = sequenceId;
 38     this.when = when;
 39 };
 40 
 41 PostTracker.prototype = {
 42 
 43     /**
 44      * Create a new Node instance and add to the LIFO size-limited queue.
 45      *
 46      * @param {Integer} sequenceId the unique identifier associated with a notification request
 47      *
 48      * @param {Date} when a Javascript Date instance the describes when the request was received
 49      */
 50     add: function (sequenceId, when) {
 51         var log = this.log.child('add');
 52         log.BEGIN(sequenceId, when);
 53         if (this.active.size() === this.maxSize) {
 54             delete this.mapping[this.active.popBack().data.sequence];
 55         }
 56         this.mapping[sequenceId] = this.active.pushFront(new PostTracker.Node(sequenceId, when));
 57         log.END();
 58     },
 59 
 60     /**
 61      * Log the end-to-end time taken to process a notification requested and receive it at a client.
 62      *
 63      * @param {Integer} sequenceId the unique identifier associated with a notification request
 64      *
 65      * @param {Date} when a Javascript Date instance the describes when the notification was received at the client.
 66      */
 67     track: function(sequenceId, when) {
 68         var log = this.log.child('track');
 69         log.BEGIN(sequenceId, when);
 70         var node = this.mapping[sequenceId];
 71         if (node) {
 72             var duration = when - node.data.when;
 73             log.debug('when.getTime:', when);
 74             log.info(sequenceId, ' receipt - duration:', duration, 'msecs');
 75         }
 76         log.END();
 77     }
 78 };
