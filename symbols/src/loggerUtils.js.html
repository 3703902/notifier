1 "use strict";
  2 
  3 /**
  4  * @fileOverview Defines some Logger utility methods.
  5  */
  6 module.exports = LoggerUtils;
  7 
  8 var log4js = require('log4js');
  9 var slice = Array.prototype.slice;
 10 
 11 /**
 12  * LoggerUtils constructor.
 13  *
 14  * @class
 15  *
 16  * Definition of helper methods that make working with Logger objects a little easier.
 17  */
 18 function LoggerUtils() {
 19 }
 20 
 21 /**
 22  * LoggerUtils prototype.
 23  */
 24 LoggerUtils.prototype = {
 25 
 26     /**
 27      * Create a new child logger whose name is derived from this one. Names are separated by '.' For instance, calling
 28      * this method on Logger object whose name is 'foo.bar' with the argument 'beep' will return a new or cached Logger
 29      * instance whose name is 'foo.bar.beep'.
 30      *
 31      * @param {String} name
 32      *   The name of the child
 33      *
 34      * @return {Logger} new or cached Logger instance with the given name
 35      */
 36     child: function(name) {
 37         return log4js.getLogger(this.category + '.' + name);
 38     },
 39 
 40     /**
 41      * Sugar method for the beginning of a block (eg. method). Emits log messages that start with the tag 'BEG'
 42      */
 43     BEGIN: function() {
 44         this.info.apply(this, ['BEG'].concat(slice.call(arguments)));
 45     },
 46 
 47     /**
 48      * Sugar method for the end of a code block (eg. method). Emits log messages that start with the tag 'END'
 49      */
 50     END: function() {
 51         this.info.apply(this, ['END'].concat(slice.call(arguments)));
 52     }
 53 };
 54 
 55 // Add the above extensions to all logger instances.
 56 
 57 var root = log4js.getDefaultLogger();
 58 var lu = LoggerUtils.prototype;
 59 
 60 root.__proto__.child = lu.child;
 61 root.__proto__.BEGIN = lu.BEGIN;
 62 root.__proto__.END = lu.END;
