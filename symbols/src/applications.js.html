1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Applications prototype and its methods.
  5  */
  6 module.exports = Applications;
  7 
  8 var express = require('express');
  9 var Model = require('model.js');
 10 var HTTPStatus = require('http-status');
 11 
 12 /**
 13  * Applications constructor.
 14  *
 15  * @class
 16  *
 17  * Applications allows for querying, updating, and deleting of application info. An application defines the credentials
 18  * to use to talk to a notification service such as APNs or WNS.
 19  *
 20  * @param {ApplicationsStore} applicationsStore the {@link ApplicationsStore} instance to rely on for persistent data
 21  * operations.
 22  */
 23 function Applications(applicationsStore) {
 24     this.log = require('./config').log('Applications');
 25     this.log.BEGIN(applicationsStore);
 26     this.applicationsStore = applicationsStore;
 27 
 28     /**
 29      * JSON schema for appId values. Contains just one value:
 30      *
 31      * - appId: the application to operate on
 32      *
 33      * @type Model
 34      */
 35     this.AppIdModel = Model.extend(
 36         {
 37             appId: {required:true, type:'string', minlength:2, maxlength:128}
 38         });
 39 
 40     /**
 41      * JSON schema for APNs credentials.
 42      *
 43      * - certificate: the public certificate to use when connecting to APNs
 44      * - privateKey: the private key to use when connecting to APNs
 45      * - passphrase: the passphrase of the privateKey
 46      * - useSandbox: true if connecting to the sandbox APNs service. Otherwise, connect to the production one.
 47      *
 48      * @type Model
 49      */
 50     this.APNsModel = Model.extend(
 51         {
 52             certificate: {required:true, type:'string', minlength:1},
 53             privateKey: {required:true, type:'string', minlength:1},
 54             passphrase: {required:true, type:'string', minlength:1},
 55             useSandbox: {required:false, type:'boolean'}
 56         });
 57 
 58     /**
 59      * JSON schema for GCM credential.
 60      *
 61      * - authorizationKey: the key obtained from GCM to use when submitting push requests
 62      *
 63      * @type Model
 64      */
 65     this.GCMModel = Model.extend(
 66         {
 67             authorizationKey: {required:true, type:'string', minlength:1}
 68         });
 69 
 70     /**
 71      * JSON schema for WNS credentials.
 72      *
 73      * - packageSID: the package SID obtained from Windows Store to use when authenticating with WNS.
 74      * - clientSecret: the client secret obtained from Windows Store to use when authenticating with WNS.
 75      *
 76      * @type Model
 77      */
 78     this.WNSModel = Model.extend(
 79         {
 80             packageSID: {required:true, type:'string', minlength:1},
 81             clientSecret: {required:true, type:'string', minlength:1}
 82         });
 83 
 84     this.log.END();
 85 }
 86 
 87 Applications.prototype = {
 88 
 89     /**
 90      * Add Express routes for the Applications API.
 91      *
 92      * @param {Express.App} app The application to route HTTP requests.
 93      */
 94     route: function(app) {
 95         var log = this.log.child('route');
 96         log.BEGIN('adding bindings');
 97         app.get('/applications/:appId', this.get.bind(this));
 98         app.post('/applications/:appId', this.set.bind(this));
 99         app.del('/applications/:appId', this.del.bind(this));
100         log.END();
101     },
102 
103     /**
104      * Obtain the current settings for a given appId.
105      *
106      * @example HTTP GET /foobar
107      *
108      * @param {Express.Request} req Describes the incoming HTTP request. Message body is ignored; the URL must contain
109      * the user ID to use, and it must adhere to the {@link Registrar#UserIdModel} model.
110      *
111      * @param {Express.Response} res HTTP response generator for the request. Outputs JSON if no error.
112      *
113      * - 200 OK if one or more registrations were found
114      * - 400 BAD REQUEST if missing or invalid query attribute(s)
115      * - 404 NOT FOUND if no templates were found for the given key attributes
116      */
117     get: function (req, res) {
118         var log = this.log.child('get');
119         var params = {appId: req.param('appId')};
120         log.BEGIN(params);
121 
122         var errors = this.AppIdModel.validate(params);
123         if (errors !== null) {
124             log.error('invalid params:',JSON.stringify(errors));
125             res.send(HTTPStatus.BAD_REQUEST);
126             return;
127         }
128 
129         var start = Date.now();
130         this.applicationsStore.get(params.appId, function (err, value) {
131             var duration = Date.now() - start;
132             if (err !== null || typeof(value) === 'undefined') {
133                 if (err) log.error('error:', err);
134                 log.info('no appId found');
135                 res.send(HTTPStatus.NOT_FOUND);
136             }
137             else {
138                 var tmp = {
139                     'credentials': value,
140                     'tableStoreDuration': duration
141                 };
142                 log.info(tmp);
143                 res.json(HTTPStatus.OK, tmp);
144             }
145             log.END(duration, 'msec');
146         });
147     },
148 
149     /**
150      * Add or update a registration for a given user.
151      *
152      * @example <code>{deviceId: "1234-56-7890", templateVersion: "3", templateLanguage: "en-US",
153      *  service: "apns",
154      *  routes:[{name: "main", token: "unique main token", secondsToLive: 86400},
155      *          {name: "alt", token: "unique alt token", secondsToLive: 86400}]}
156      * </code>
157      *
158      * @param {Express.Request} req Describes the incoming HTTP request. Message body must be JSON that follows the
159      * {@link Registrar#RegistrationModel} model.
160      *
161      * @param {Express.Response} res HTTP response generator for the request. Outputs JSON if no error.
162      *
163      * - 200 OK: successfully processed request
164      * - 400 BAD REQUEST: missing or invalid attribute(s)
165      */
166     set: function (req, res) {
167         var self = this;
168         var log = self.log.child('set');
169         log.BEGIN();
170 
171         var params = req.body;
172         params.userId = req.param('userId');
173         log.info('params:', params);
174 
175         var errors = self.RegistrationModel.validate(params);
176         if (errors !== null) {
177             log.error('invalid params:', JSON.stringify(params));
178             self.monitorManager.post(params.userId, 'invalid registration: ' + JSON.stringify(params));
179             res.send(HTTPStatus.BAD_REQUEST);
180             return;
181         }
182 
183         var start = Date.now();
184         self.registrationStore.set(params, function (err, registrationEntity) {
185             var duration = Date.now() - start;
186             if (err) {
187                 log.error('RegistrationStore.set error:', err);
188                 self.monitorManager.post(params.userId, 'failed to store registration');
189                 res.send(HTTPStatus.BAD_REQUEST);
190             }
191             else {
192                 var tmp = {
193                     'tableStoreDuration': duration
194                 };
195                 log.info(tmp);
196                 self.monitorManager.post(params.userId, 'registration: ' + JSON.stringify(params));
197                 res.json(HTTPStatus.OK, tmp);
198             }
199             log.END(params.userId, duration, 'msec');
200         });
201     },
202 
203     /**
204      * Delete a registration or all registrations for a given user. If a JSON payload exits with a deviceId
205      * attribute, only delete the one entity matching the deviceId value. Otherwise, delete all registrations for
206      * the user.
207      *
208      * @example <code>{}</code>
209      *
210      * @example <code>{deviceId: "1234-56-7890"}</code>
211      *
212      * @param {Express.Request} req Describes the incoming HTTP request. Message body must be JSON that follows the
213      * {@link Registrar#DeleteKeyModel} model.
214      *
215      * @param {Express.Response} res HTTP response generator for the request. Outputs JSON if no error.
216      *
217      * - 204 NO CONTENT: found and deleted the registration(s)
218      * - 400 BAD REQUEST: missing or invalid attribute(s)
219      * - 404 NOT FOUND: unable to locate any registration to delete
220      */
221     del: function (req, res) {
222         var log = this.log.child('del');
223         log.BEGIN('body:', req.body);
224 
225         var params = {
226             userId: req.param('userId')
227         };
228 
229         var deviceId = req.param('deviceId');
230         if (typeof deviceId !== 'undefined') {
231             params.deviceId = deviceId;
232         }
233 
234         log.info('params:', params);
235         var errors = this.DeleteKeyModel.validate(params);
236         if (errors !== null) {
237             log.error('invalid params:', JSON.stringify(errors));
238             res.send(HTTPStatus.BAD_REQUEST);
239             return;
240         }
241 
242         var start = Date.now();
243         function callback(err) {
244             var duration = Date.now() - start;
245             if (err) {
246                 log.error('RegistrationStore.del error:', err);
247                 res.send(HTTPStatus.NOT_FOUND);
248             }
249             else {
250                 self.monitorManager.post(params.userId, 'removed registration(s)');
251                 res.send(HTTPStatus.NO_CONTENT);
252             }
253             log.END(params.userId, deviceId, duration, 'msec');
254         }
255 
256         if (params.deviceId === undefined) {
257             this.registrationStore.delAll(params.userId, callback);
258         }
259         else {
260             this.registrationStore.del(params.userId, params.deviceId, callback);
261         }
262     }
263 };
