1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Deque prototype and its methods.
  5  */
  6 module.exports = Deque;
  7 
  8 /**
  9  * Deque constructor.
 10  *
 11  * @class
 12  *
 13  * An Deque object acts much like the C++ std::deque container. Provides operations to add/remove objects from the
 14  * start and the end of the containuer.
 15  */
 16 function Deque() {
 17     this._head = new Deque.Node();
 18     this._tail = new Deque.Node();
 19     this._head._next = this._tail;
 20     this._tail._prev = this._head;
 21     this._size = 0;
 22 }
 23 
 24 /**
 25  * Deque.Node constructor.
 26  *
 27  * @class
 28  *
 29  * A Node object simple maintains links to a Deque double linked list.
 30  */
 31 Deque.Node = function(data) {
 32     this.data = data;
 33     this._prev = null;
 34     this._next = null;
 35 };
 36 
 37 Deque.Node.prototype = {
 38 
 39     /**
 40      * Remove this node from the queue.
 41      */
 42     extricate: function () {
 43         this._prev._next = this._next;
 44         this._next._prev = this._prev;
 45     }
 46 };
 47 
 48 /**
 49  * Deque prototype.
 50  *
 51  * Defines the methods available to a Deque instance.
 52  */
 53 Deque.prototype = {
 54 
 55     /**
 56      * Determine if the container has no elements.
 57      *
 58      * @return true if so
 59      */
 60     empty: function () {
 61         return this._head._next === this._tail;
 62     },
 63 
 64     /**
 65      * Obtain the number of elements held in the container.
 66      *
 67      * @return {Number} current container capacity
 68      */
 69     size: function () {
 70         return this._size;
 71     },
 72 
 73     backNode: function () {
 74         if (this.empty()) {
 75             throw new Error("backNode called on empty deque");
 76         }
 77         return this._tail._prev;
 78     },
 79 
 80     back: function () {
 81         return this.backNode().data;
 82     },
 83 
 84     frontNode: function () {
 85         if (this.empty()) {
 86             throw new Error("frontNode called on empty deque");
 87         }
 88         return this._head._next;
 89     },
 90 
 91     front: function () {
 92         return this.frontNode().data;
 93     },
 94 
 95     /**
 96      * Remove a node from the queue.
 97      *
 98      * @param {Node} node the node to remove
 99      */
100     removeNode: function(node) {
101         node.extricate();
102         --this._size;
103     },
104 
105     /**
106      * Add an existing Node instance to the back of the queue.
107      *
108      * @param {Node} node Node instance to add to the queue
109      */
110     pushBackNode: function (node) {
111         node._prev = this._tail._prev;
112         node._prev._next = node;
113         node._next = this._tail;
114         this._tail._prev = node;
115         ++this._size;
116         return node;
117     },
118 
119     /**
120      * Add a new Node instance to the back of the queue.
121      *
122      * @param {Object} data value to add to the queue
123      */
124     pushBack: function (data) {
125         var node = new Deque.Node(data);
126         node._prev = this._tail._prev;
127         node._prev._next = node;
128         node._next = this._tail;
129         this._tail._prev = node;
130         ++this._size;
131         return node;
132     },
133 
134     /**
135      * Remove a Node instance from the back of the queue.
136      *
137      * @returns {Node} last Node in the queue
138      */
139     popBackNode: function () {
140         if (this.empty()) {
141             throw new Error("popBack called on empty deque");
142         }
143         --this._size;
144         var node = this._tail._prev;
145         this._tail._prev = node._prev;
146         node._prev._next = this._tail;
147         return node;
148     },
149 
150     popBack: function() {
151         return this.popBackNode().data;
152     },
153 
154     /**
155      * Add an existing Node instance to the front of the queue.
156      *
157      * @param {Node} node Node instance to add to the queue
158      */
159     pushFrontNode: function (node) {
160         node._next = this._head._next;
161         node._next._prev = node;
162         node._prev = this._head;
163         this._head._next = node;
164         ++this._size;
165         return node;
166     },
167 
168     /**
169      * Add a new Node instance to the front of the queue.
170      *
171      * @param {Node} data value to add to the queue
172      */
173     pushFront: function (data) {
174         var node = new Deque.Node(data);
175         node._next = this._head._next;
176         node._next._prev = node;
177         node._prev = this._head;
178         this._head._next = node;
179         ++this._size;
180         return node;
181     },
182 
183     /**
184      * Remove a Node instance from the front of the queue.
185      *
186      * @returns {Object} first Node in the queue
187      */
188     popFrontNode: function () {
189         if (this.empty()) {
190             throw new Error("popFront called on empty deque");
191         }
192         --this._size;
193         var node = this._head._next;
194         this._head._next = node._next;
195         node._next._prev = this._head;
196         return node;
197     },
198 
199     popFront: function() {
200         return this.popFrontNode().data;
201     }
202 };
