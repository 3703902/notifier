1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Registrar prototype and its methods.
  5  */
  6 module.exports = Registrar;
  7 
  8 var express = require('express');
  9 var Model = require('model.js');
 10 var HTTPStatus = require('http-status');
 11 
 12 /**
 13  * Registrar constructor.
 14  *
 15  * @class
 16  *
 17  * A registrar allows for querying, updating, and deleting of registrations for a given user ID.
 18  *
 19  * @param {RegistrationStore} registrationStore the RegistrationStore instance to rely on for persistent data
 20  * operations.
 21  */
 22 function Registrar(registrationStore) {
 23     this.log = require('./config').log('Registrar');
 24     this.log.BEGIN(registrationStore);
 25 
 26     this.registrationStore = registrationStore;
 27 
 28     /**
 29      * JSON schema for userId values. Contains just one value:
 30      *
 31      * - userId: the user to operate on
 32      *
 33      * @type Model
 34      */
 35     this.UserIdModel = Model.extend(
 36         {
 37             userId: {required:true, type:'string', minlength:2, maxlength:128}
 38         });
 39 
 40     /**
 41      * JSON schema for route definitions. Attributes:
 42      *
 43      * - name: the name of the route. Only receives notifications from templates with a matching name value.
 44      * - token: the OS-specific value associated with the device to notify
 45      * - secondsToLive: the number of seconds this registration route will be valid
 46      *
 47      * @type Model
 48      */
 49     this.RouteModel = Model.extend(
 50         {
 51             name: {required:true, type:'string', minlength:1, maxlength:128},
 52             token: {required:true, type:'string', minlength:1},
 53             secondsToLive: {required:true, type:'integer', min:1}
 54         });
 55 
 56     /**
 57      * JSON schema for registration definitions. Attributes:
 58      *
 59      * - registrationId: the unique identifier associated with the device being registered. This must be unique across
 60      *   all devices, but it should be constant for the same device.
 61      * - templateVersion: the version to look for when searching for templates to use for this device
 62      * - templateLanguage: the language to look for when searching for templates to use for this device
 63      * - service: identifier of the OS-specific service that handles notification delivery ("wns", "apns", "gcm")
 64      * - routes: an array of one or more {@link Registrar#RouteModel} objects that define the registration delivery
 65      *           paths for the device
 66      *
 67      * @type Model
 68      */
 69     this.RegistrationModel = this.UserIdModel.extend(
 70         {
 71             registrationId: {required:true, type:'string', minlength:1, maxlength:128},
 72             templateVersion: {required:true, type:'string', minlength:1, maxlength:10},
 73             templateLanguage: {required:true, type:'string', minlength:2, maxlength:10},
 74             service: {required:true, type:'string', minlength:1, maxlength:10},
 75             routes: {required:true, type:'routeArray', minlength:1}
 76         },
 77         {
 78             types:{
 79                 routeArray:function (value) {
 80                     value = Model.types.array(value);
 81                     if (value === null) return value;
 82                     for (var index in value) {
 83                         var z = this.RouteModel.validate(value[index]);
 84                         if (z !== null) return null;
 85                     }
 86                     return value;
 87                 }.bind(this)
 88             }
 89         });
 90 
 91     /**
 92      * JSON schema for delete parameter definitions. Attributes:
 93      *
 94      * - registrationId: the unique identifier associated with the device to be removed. See
 95      *   {@link Registrar#RegistrationModel} for additional information.
 96      *
 97      * @type Model
 98      */
 99     this.DeleteKeyModel = this.UserIdModel.extend(
100         {
101             registrationId: {type:'string', minlength:1, maxlength:128}
102         });
103 
104     this.log.END();
105 }
106 
107 /**
108  * Registrar prototype.
109  *
110  * Defines the methods available to a Registrar instance.
111  */
112 Registrar.prototype = {
113 
114     /**
115      * Add Express routes for the Registrar API.
116      *
117      * @param {Express.App} app the application to route
118      */
119     route: function(app) {
120         var log = this.log.child('route');
121         log.BEGIN('adding bindings');
122         app.get('/registrations/:userId', this.get.bind(this));
123         app.post('/registrations/:userId', this.set.bind(this));
124         app.del('/registrations/:userId', this.del.bind(this));
125         log.END();
126     },
127 
128     /**
129      * Obtain the current registrations for a given user. Parameters:
130      *
131      * - userId: the user to look for (found in the URL)
132      *
133      * HTTP Responses:
134      *
135      * - 400 BAD REQUEST: missing or invalid query attribute(s)
136      * - 404 NOT FOUND: no templates were found for the given key attributes
137      * - 200 OK: one or more registrations were found
138      *
139      * @param req Contents of the incoming request.
140      *
141      * @param res Response generator for the request. Generates JSON output if there are registrations
142      */
143     get: function (req, res) {
144         var log = this.log.child('get');
145         log.BEGIN();
146 
147         var params = {userId: req.param('userId')};
148         log.info('params:', params);
149 
150         var errors = this.UserIdModel.validate(params);
151         if (errors !== null) {
152             log.error('invalid params:',JSON.stringify(errors));
153             res.send(HTTPStatus.BAD_REQUEST);
154             return;
155         }
156 
157         var start = Date.now();
158         this.registrationStore.get(params.userId, null, function (err, regs) {
159             var duration = Date.now() - start;
160             if (err !== null || typeof(regs) === 'undefined' || regs.length === 0) {
161                 if (err) log.error('error:', err);
162                 log.info('no registrations found');
163                 res.send(HTTPStatus.NOT_FOUND);
164             }
165             else {
166                 var tmp = {
167                     'registrations': regs,
168                     'tableStoreDuration': duration
169                 };
170                 log.info(tmp);
171                 res.json(HTTPStatus.OK, tmp);
172             }
173             log.END(duration, 'msec');
174         });
175     },
176 
177     /**
178      * Add or update a registration for a given user. Parameters:
179      *
180      * - userId: the user to look for (found in the URL)
181      * - registrationId: the registration to add/update.
182      * - templateVersion: the template version for this registration.
183      * - templateLanguage: the template language for this registration.
184      * - service: the notification service for this registration.
185      * - routes: array of one or more route dicts with keys:
186      *   - name: the unique name of the route for this registration.
187      *   - token: service-specific way to reach the user for notifications.
188      *   - secondsToLive: the number of seconds into the future the token is valid.
189      *
190      * HTTP Responses:
191      *
192      * - 400 BAD REQUEST: missing or invalid template attribute(s)
193      * - 200 OK: one or moret templates were found
194      *
195      * @param req Contents of the incoming request.
196      *
197      * @param res Response generator for the request.
198      */
199     set: function (req, res) {
200         var self = this;
201         var log = this.log.child('set');
202         log.BEGIN();
203 
204         var params = req.body;
205         params.userId = req.param('userId');
206         log.info('params:', params);
207 
208         var errors = this.RegistrationModel.validate(params);
209         if (errors !== null) {
210             log.error('invalid params:', JSON.stringify(params));
211             res.send(HTTPStatus.BAD_REQUEST);
212             return;
213         }
214 
215         var start = Date.now();
216         this.registrationStore.set(params, function (err, registrationEntity) {
217             var duration = Date.now() - start;
218             if (err) {
219                 log.error('RegistrationStore.set error:', err);
220                 res.send(HTTPStatus.BAD_REQUEST);
221             }
222             else {
223                 var tmp = {
224                     'tableStoreDuration': duration
225                 };
226                 log.info(tmp);
227                 res.json(HTTPStatus.OK, tmp);
228             }
229             log.END(params.userId, duration, 'msec');
230         });
231     },
232 
233     /**
234      * Delete a registration or all registrations for a given user. If a JSON
235      * payload exits with a registrationId attribute, only delete the one entity
236      * matching the registrationId value. Otherwise, delete all registrations for the
237      * user.
238      *
239      * - userId: the user to look for (found in the URL)
240      * - registrationId (optional): the registration to delete
241      *
242      * HTTP Responses:
243      *
244      * - 400 BAD REQUEST: missing or invalid template attribute(s)
245      * - 404 NOT FOUND: unable to locate any registration to delete
246      * - 204 NO CONTENT: found and delete the registration(s)
247      *
248      * @param req Contents of the incoming request.
249      *
250      * @param res Response generator for the request.
251      */
252     del: function (req, res) {
253         var log = this.log.child('del');
254         log.BEGIN('body:', req.body);
255 
256         var params = {
257             userId: req.param('userId')
258         };
259 
260         var registrationId = req.param('registrationId');
261         if (typeof registrationId !== 'undefined') {
262             params.registrationId = registrationId;
263         }
264 
265         log.info('params:', params);
266         var errors = this.DeleteKeyModel.validate(params);
267         if (errors !== null) {
268             log.error('invalid params:', JSON.stringify(errors));
269             res.send(HTTPStatus.BAD_REQUEST);
270             return;
271         }
272 
273         var start = Date.now();
274         function callback(err) {
275             var duration = Date.now() - start;
276             if (err) {
277                 log.error('RegistrationStore.del error:', err);
278                 res.send(HTTPStatus.NOT_FOUND);
279             }
280             else {
281                 res.send(HTTPStatus.NO_CONTENT);
282             }
283             log.END(params.userId, registrationId, duration, 'msec');
284         }
285 
286         if (params.registrationId === undefined) {
287             this.registrationStore.delAll(params.userId, callback);
288         }
289         else {
290             this.registrationStore.del(params.userId, params.registrationId, callback);
291         }
292     }
293 };
