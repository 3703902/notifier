1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Registrar prototype and its methods.
  5  */
  6 module.exports = Registrar;
  7 
  8 var Model = require('model.js');
  9 var HTTPStatus = require('http-status');
 10 
 11 /**
 12  * Registrar constructor.
 13  *
 14  * @class
 15  *
 16  * A registrar allows for querying, updating, and deleting of registrations for a given user ID.
 17  */
 18 function Registrar(registrationStore) {
 19     this.log = require('./config').log('registrar');
 20     this.registrationStore = registrationStore;
 21 
 22     /**
 23      * JSON schema for userId values
 24      * @type Model
 25      */
 26     this.UserIdModel = Model.extend(
 27         {
 28             userId: {required:true, type:'string', minlength:2, maxlength:128}
 29         });
 30 
 31     /**
 32      * JSON schema for route definitions.
 33      * @type Model
 34      */
 35     this.RouteModel = Model.extend(
 36         {
 37             name: {required:true, type:'string', minlength:1, maxlength:128},
 38             token: {required:true, type:'string', minlength:1},
 39             secondsToLive: {required:true, type:'integer', min:1}
 40         });
 41 
 42     /**
 43      * JSON schema for registration definitions.
 44      * @type Model
 45      */
 46     this.RegistrationModel = this.UserIdModel.extend(
 47         {
 48             registrationId: {required:true, type:'string', minlength:1, maxlength:128},
 49             templateVersion: {required:true, type:'string', minlength:1, maxlength:10},
 50             templateLanguage: {required:true, type:'string', minlength:2, maxlength:10},
 51             service: {required:true, type:'string', minlength:1, maxlength:10},
 52             routes: {required:true, type:'routeArray', minlength:1}
 53         },
 54         {
 55             types:{
 56                 routeArray:function (value) {
 57                     value = Model.types.array(value);
 58                     if (value === null) return value;
 59                     for (var index in value) {
 60                         var z = this.RouteModel.validate(value[index]);
 61                         if (z !== null) return null;
 62                     }
 63                     return value;
 64                 }.bind(this)
 65             }
 66         });
 67 
 68     /**
 69      * JSON schema for delete parameter definitions.
 70      * @type Model
 71      */
 72     this.DeleteKeyModel = this.UserIdModel.extend(
 73         {
 74             registrationId: {type:'string', minlength:1, maxlength:128}
 75         });
 76 }
 77 
 78 /**
 79  * Registrar prototype.
 80  *
 81  * Defines the methods available to a Registrar instance.
 82  */
 83 Registrar.prototype = {
 84 
 85     /**
 86      * Obtain the current registrations for a given user.
 87      *
 88      * @param req
 89      *   Contents of the incoming request
 90      *
 91      * @param res
 92      *   Response generator for the request. Generates JSON output if the
 93      *   client requested JSON.
 94      */
 95     getRegistrations: function (req, res) {
 96         var log = this.log.child('getRegistrations');
 97         log.BEGIN();
 98 
 99         var params = {userId: req.param('userId')};
100         log.info('params:', params);
101 
102         var errors = this.UserIdModel.validate(params);
103         if (errors !== null) {
104             log.error('invalid params:',JSON.stringify(errors));
105             res.send(HTTPStatus.BAD_REQUEST);
106             return;
107         }
108 
109         var start = Date.now();
110         this.registrationStore.getRegistrations(params.userId, null, function (err, regs) {
111             var duration = Date.now() - start;
112             if (err !== null || typeof(regs) === 'undefined' || regs.length === 0) {
113                 if (err) log.error('error:', err);
114                 log.info('no registrations found');
115                 res.send(HTTPStatus.NOT_FOUND);
116             }
117             else {
118                 var tmp = {
119                     'registrations': regs,
120                     'tableStoreDuration': duration
121                 };
122                 log.info(tmp);
123                 res.json(HTTPStatus.OK, tmp);
124             }
125             log.END(duration, 'msec');
126         });
127     },
128 
129     /**
130      * Add or update a registration for a given user.
131      *
132      * @param req
133      *   Contents of the incoming request. Expects a body in JSON format.
134      *   - req.params.userId: the user to look for
135      *   - req.body.registrationId: the registration to add/update.
136      *   - req.body.templateVersion: the template version for this registration.
137      *   - req.body.templateLanguage: the template language for this registration.
138      *   - req.body.service: the notification service for this registration.
139      *   - req.body.routes: array of one or more route dicts with keys:
140      *     - name: the unique name of the route for this registration.
141      *     - token: service-specific way to reach the user for notifications.
142      *     - secondsToLive: the number of seconds into the future the token is valid.
143      *
144      * @param res
145      *   Response generator for the request. Generates JSON output if the
146      *   client requested JSON.
147      */
148     addRegistration: function (req, res) {
149         var self = this;
150         var log = this.log.child('addRegistration');
151         log.BEGIN('req:', req.body);
152 
153         var params = req.body;
154         params.userId = req.param('userId');
155         log.info('params:', params);
156 
157         var errors = this.RegistrationModel.validate(params);
158         if (errors !== null) {
159             log.error('invalid params:', JSON.stringify(params));
160             res.send(HTTPStatus.BAD_REQUEST);
161             return;
162         }
163 
164         var start = Date.now();
165         this.registrationStore.updateRegistration(params, function (err, registrationEntity)
166         {
167             var duration = Date.now() - start;
168             if (err) {
169                 log.error('RegistrationStore.updateRegistration error:', err);
170                 res.send(HTTPStatus.BAD_REQUEST);
171             }
172             else {
173                 var tmp = {
174                     'tableStoreDuration': duration
175                 };
176                 log.info(tmp);
177                 res.json(HTTPStatus.OK, tmp);
178             }
179             log.END(params.userId, duration, 'msec');
180         });
181     },
182 
183     /**
184      * Delete a registration or all registrations for a given user. If a JSON
185      * payload exits with a registrationId attribute, only delete the one entity
186      * matching the registrationId value. Otherwise, delete all registrations for the
187      * user.
188      *
189      * @param req
190      *   Contents of the incoming request.
191      *   - req.params.userId: the user to look for
192      *   - req.body.registrationId (optional): the registration to delete
193      *
194      * @param res
195      *   Response generator for the request. Generates JSON output if the
196      *   client requested JSON.
197      */
198     deleteRegistration: function (req, res) {
199         var log = this.log.child('deleteRegistration');
200         log.BEGIN();
201 
202         var params = {
203             userId: req.param('userId')
204         };
205 
206         var registrationId = req.param('registrationId');
207         if (typeof registrationId !== 'undefined') {
208             params.registrationId = registrationId;
209         }
210 
211         log.info('params:', params);
212         var errors = this.DeleteKeyModel.validate(params);
213         if (errors !== null) {
214             log.error('invalid params:', JSON.stringify(errors));
215             res.send(HTTPStatus.BAD_REQUEST);
216             return;
217         }
218 
219         var start = Date.now();
220         function callback(err) {
221             var duration = Date.now() - start;
222             if (err) {
223                 log.error('RegistrationStore.deleteRegistrations error:', err);
224                 res.send(HTTPStatus.NOT_FOUND);
225             }
226             else {
227                 res.send(HTTPStatus.NO_CONTENT);
228             }
229             log.END(params.userId, registrationId, duration, 'msec');
230         }
231 
232         if (params.registrationId === undefined) {
233             this.registrationStore.deleteAllRegistrations(params.userId, callback);
234         }
235         else {
236             this.registrationStore.deleteRegistration(params.userId, params.registrationId, callback);
237         }
238     }
239 };
