1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the PayloadGenerator prototype and its methods.
  5  */
  6 module.exports = PayloadGenerator;
  7 
  8 var XRegExp = require('xregexp').XRegExp;
  9 var config = require('./config');
 10 var filters = require('./filters');
 11 
 12 /**
 13  * PayloadGenerator constructor.
 14  *
 15  * @class
 16  *
 17  * A PayloadGenerator transforms notification template into notification payloads, replacing any placeholdes with live
 18  * data.
 19  */
 20 function PayloadGenerator() {
 21     this.log = config.log('PayloadGenerator');
 22 
 23     /**
 24      * The regular expression that matches a placeholder in a template.
 25      * @type XRegExp
 26      */
 27     this.regexp =  XRegExp(config.templates_placeholder_re);
 28     this.filters = filters;
 29 }
 30 
 31 /**
 32  * PayloadGenerator prototype.
 33  *
 34  * Defines the methods available to a PayloadGenerator instance.
 35  */
 36 PayloadGenerator.prototype = {
 37 
 38     /**
 39      * Transform a notification template into a payload by substitution placeholders with given substitution values.
 40      *
 41      * @param {Object} content the template to process
 42      *
 43      * @param {Object} substitutions mapping of placeholder names to substitution values
 44      */
 45     transform: function(content, substitutions) {
 46         var log = this.log.child('transform');
 47         log.BEGIN(content, substitutions);
 48 
 49         if (! substitutions) {
 50             substitutions = {};
 51         }
 52 
 53         while (1) {
 54             var match = this.regexp.exec(content);
 55             if (match === null) break;
 56             log.debug('found', match);
 57             var value = substitutions[match[1]];
 58             if (typeof value === 'undefined') {
 59                 value = match[3];
 60                 if (typeof value === 'undefined') {
 61                     log.debug('no default value defined');
 62                     value = '';
 63                 }
 64             }
 65             log.debug('substituting with', value);
 66             content = content.replace(match[0], value);
 67         }
 68 
 69         return content;
 70     }
 71 };
