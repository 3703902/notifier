1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the PayloadGenerator prototype and its methods.
  5  */
  6 module.exports = PayloadGenerator;
  7 
  8 var XRegExp = require('xregexp').XRegExp;
  9 var config = require('./config');
 10 
 11 /**
 12  * PayloadGenerator constructor.
 13  *
 14  * @class
 15  *
 16  * A PayloadGenerator transforms notification template into notification payloads, replacing any placeholdes with live
 17  * data.
 18  */
 19 function PayloadGenerator() {
 20     this.log = config.log('payloadGenerator');
 21 
 22     /**
 23      * The regular expression that matches a placeholder in a template.
 24      * @type XRegExp
 25      */
 26     this.regexp =  XRegExp(config.templates_placeholder_re);
 27 }
 28 
 29 /**
 30  * PayloadGenerator prototype.
 31  *
 32  * Defines the methods available to a PayloadGenerator instance.
 33  */
 34 PayloadGenerator.prototype = {
 35 
 36     /**
 37      * Transform a notification template into a payload by substitution placeholders with given substitution values.
 38      *
 39      * @param {String} template the template to process
 40      *
 41      * @param {Object} substitutions mapping of placeholder names to substitution values
 42      *
 43      * @return {String} notification payload
 44      */
 45     transform: function(template, substitutions) {
 46         var log = this.log.child('transform');
 47 
 48         log.BEGIN(template, substitutions);
 49 
 50         if (! substitutions) {
 51             substitutions = {};
 52         }
 53 
 54         while (1) {
 55             var match = this.regexp.exec(template);
 56             if (match === null) break;
 57             log.debug('found', match[0]);
 58             var value = substitutions[match[1]];
 59             if (value === undefined) {
 60                 log.debug('not in substitutions');
 61                 value = match[3];
 62                 if (value === undefined) {
 63                     log.debug('no default value defined');
 64                     value = '';
 65                 }
 66             }
 67             log.debug('substituting with', value);
 68             template = template.replace(match[0], value);
 69         }
 70 
 71         return template;
 72     }
 73 };
