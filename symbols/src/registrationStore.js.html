 12 /**
 13  * RegistrationStore constructor.
 14  *
 15  * @class RegistrationStore
 16  */
 17 function RegistrationStore(tableName, callback) {
 18     var self = this;
 19     var log = this.log = require('./config').log('RegistrationStore');
 20     log.BEGIN(tableName);
 21     if (tableName === undefined) {
 22         tableName = config.registrations_table_name;
 23     }
 24 
 25     this.tableName = tableName;
 26     this.store = azure.createTableService();
 27 
 28     var checkCreateTableIfNotExists = function (err, b, c) {
 29         if (err) {
 30             log.error(err);
 31             if (err.statusCode === 409) {
 32                 Q.delay(1000)   // Wait for one second and try again.
 33                     .then(function () {
 34                         self.store.createTableIfNotExists(tableName, checkCreateTableIfNotExists);
 35                     });
 36                 return;
 37             }
 38         }
 39 
 40         log.END(err, b, c);
 41         if (callback) callback(err);
 42     };
 43 
 44     this.store.createTableIfNotExists(tableName, checkCreateTableIfNotExists);
 45 }

 47 /**
 48  * Store prototype.
 49  *
 50  * Defines the methods available to a Store instance.
 51  */
 52 RegistrationStore.prototype = {
 53 
 54     makePartitionKey: function (value) {
 55         return new Buffer(value).toString('base64');
 56     },
 57 
 58     makeRowKey: function (value) {
 59         return new Buffer(value).toString('base64');
 60     },
 61 
 62     /**
 63      * Obtain all registration entities for a given user.
 64      *
 65      * @param {String} partitionKey the partition where the registrations reside
 66      * @param {function(error, found)} callback method to invoke when the query is done
 67      *
 68      * @private
 69      */
 70     _getEntities: function (partitionKey, callback) {
 71         var log = this.log.child('_getEntities');
 72         log.BEGIN(partitionKey);
 73         var query = azure.TableQuery.select()
 74                                     .from(this.tableName)
 75                                     .where('PartitionKey eq ?', partitionKey);
 76         this.store.queryEntities(query, callback);
 77         log.END();
 78     },

 80     /**
 81      * Delete a specific registration for a given user.
 82      *
 83      * @param {String} partitionKey the partition where the registration resides
 84      *
 85      * @param {String} rowKey the row key of the registration to delete
 86      *
 87      * @param {function(err)} callback the function to call when done
 88      *
 89      * @private
 90      */
 91     _delEntity: function (partitionKey, rowKey, callback) {
 92         var log = this.log.child('_delEntity');
 93         log.BEGIN('partitionKey:', partitionKey, 'rowKey:', rowKey);
 94         var entity = {
 95             PartitionKey: partitionKey,
 96             RowKey: rowKey
 97         };
 98         this.store.deleteEntity(this.tableName, entity, function (err) {
 99             callback(err);
100             log.END(err);
101         });
102     },

104     /**
105      * Create a JSON object from a registration table store entity.
106      *
107      * @param {Object} registrationEntity the entity to convert
108      * @param {function(reg)} filter a filter o invoke on each registration object to see if it is wanted in the results
109      * @return {Object} registration object or null.
110      */
111     _registrationFromEntity: function (registrationEntity, filter) {
112         var log = this.log.child('_registrationFromEntity');
113         log.BEGIN(registrationEntity);
114 
115         var registration = {
116             deviceId: new Buffer(registrationEntity.RowKey, 'base64').toString('ascii'),
117             templateVersion: registrationEntity.TemplateVersion.substr(1),
118             templateLanguage: registrationEntity.TemplateLanguage,
119             service: registrationEntity.Service,
120             expiration: registrationEntity.Expiration,
121             routes: null
122         };
123 
124         var now = new Date();
125         now = now.toISOString();
126 
127         var routes = [];
128         try {
129             routes = JSON.parse(registrationEntity.Routes);
130         }
131         catch (err) {
132             log.error('failed to parse registrationEntity.Routes:', registrationEntity.Routes);
133         }
134 
135         log.debug('routes:', routes);
136 
137         var valid = [];
138         for (var each in routes) {
139             var route = routes[each];
140             log.debug('route:', route);
141             if (route.Expiration > now) {
142                 var blah = {
143                     "name": route.Name,
144                     "token": route.Token,
145                     "expiration": route.Expiration
146                 };
147 
148                 log.debug('blah:', blah);
149                 if (filter.passed(registration, blah)) {
150                     log.debug('using route', blah);
151                     valid.push(blah);
152                 }
153             }
154             else {
155                 log.info('route is no longer active:', blah);
156             }
157         }
158 
159         if (valid.length === 0) {
160             registration = null;
161         }
162         else {
163             log.debug('valid routes:', valid);
164             registration.routes = valid;
165         }
166 
167         log.END(registration);
168         return registration;
169     },

171     /**
172      * Obtain the registrations for a given user, subject to them passing the given filter.
173      *
174      * @param {String} userId the user to query
175      * @param {Object} filter attributes to use to pass only certain registrations
176      * @param {function(err, registrations)} callback methoed to invoke when the fetching is done
177      */
178     get: function (userId, filter, callback) {
179         var self = this;
180         var log = self.log.child('get');
181         log.BEGIN(userId, filter);
182 
183         var partitionKey = this.makePartitionKey(userId);
184         this._getEntities(partitionKey, function (err, regs) {
185             if (err !== null) {
186                 log.error('_getEntities error:', err);
187                 callback(err, null);
188                 return;
189             }
190 
191             log.error('found:', regs);
192 
193             var registrations = [];
194 
195             if (regs.length === 0) {
196                 log.info('no registrations found');
197                 callback(null, registrations);
198                 return;
199             }
200 
201             filter = new RegistrationFilter(filter);
202 
203             var now = new Date();
204             now = now.toISOString();
205 
206             function delEntityCallback (err) {
207                 if (err) log.error('_delEntity error:', err);
208             }
209 
210             for (var index in regs) {
211                 var registrationEntity = regs[index];
212                 if (registrationEntity.Expiration <= now) {
213                     log.info('registration has expired:', registrationEntity.RowKey);
214                     self._delEntity(partitionKey, registrationEntity.RowKey, delEntityCallback);
215                 }
216                 else {
217                     var r = self._registrationFromEntity(registrationEntity, filter);
218                     if (r) registrations.push(r);
219                 }
220             }
221             callback(null, registrations);
222             log.END();
223         });
224     },

226     /**
227      * Update a specific registration for a given user.
228      *
229      * @param {Object} registration the registration settings to use or update
230      * @param {Function} callback nethod to invoke when the query is done
231      */
232     set: function (registration, callback) {
233         var self = this;
234         var log = self.log.child('set');
235         log.BEGIN(registration);
236 
237         var partitionKey = this.makePartitionKey(registration.userId);
238         log.debug('partitionKey:', partitionKey);
239 
240         var rowKey = this.makeRowKey(registration.deviceId);
241         log.debug('rowKey:', rowKey);
242 
243         self.store.queryEntity(self.tableName, partitionKey, rowKey, function (err, found) {
244             log.info('queryEntity results:', err, found);
245 
246             var registrationEntity = {
247                 'PartitionKey': partitionKey,
248                 'RowKey': rowKey,
249                 'TemplateLanguage': registration.templateLanguage,
250                 'TemplateVersion': 'v' + registration.templateVersion,
251                 'Service': registration.service,
252                 'Expiration': null,
253                 'Routes': null
254             };
255 
256             var now = Date.now();
257             var oldest = now;
258             var routes = registration.routes;
259 
260             for (var index in registration.routes) {
261                 var route = routes[index];
262                 var expiration = now + route.secondsToLive * 1000;
263                 if (oldest < expiration) {
264                     oldest = expiration;
265                 }
266                 route = {
267                     'Name': route.name,
268                     'Token': route.token,
269                     'Expiration': (new Date(expiration)).toISOString()
270                 };
271 
272                 log.debug('new route:', route);
273                 routes[index] = route;
274             }
275 
276             registrationEntity.Expiration = (new Date(oldest)).toISOString();
277             registrationEntity.Routes = JSON.stringify(routes);
278             if (! found) {
279                 self.store.insertEntity(self.tableName, registrationEntity, callback);
280             }
281             else {
282                 self.store.updateEntity(self.tableName, registrationEntity, callback);
283             }
284 
285             log.END();
286 	});
287     },

289     /**
290      * Delete a specific registration for a given user.
291      *
292      * @param userId
293      *   The user to look for.
294      *
295      * @param deviceId
296      *   The device entry to delete.
297      *
298      * @param callback
299      *   Method to invoke when the deletion is done.
300      *   - err: if not null, definition of the error that took place
301      *
302      * @private
303      */
304     del: function (userId, deviceId, callback) {
305         var log = this.log.child('del');
306         log.BEGIN(userId, deviceId);
307         this._delEntity(this.makePartitionKey(userId), this.makeRowKey(deviceId), function (err) {
308             callback(err);
309             log.END(err);
310         });
311     },

313     /**
314      * Delete a specific registration for a given user.
315      *
316      * @param userId
317      *   The user to look for.
318      *
319      * @param deviceId
320      *   The registration to delete.
321      *
322      * @param callback
323      *   Method to invoke when the deletion is done.
324      *   - err: if not null, definition of the error that took place
325      *
326      * @private
327      */
328     delRoute: function (userId, deviceId, routeNameOrToken) {
329         var log = this.log.child('delRoute');
330         log.BEGIN(userId, deviceId, routeToken);
331 
332         var partitionKey = this.makePartitionKey(userId);
333         log.debug('partitionKey:', partitionKey);
334 
335         var rowKey = this.makeRowKey(deviceId);
336         log.debug('rowKey:', rowKey);
337 
338         self.store.queryEntity(self.tableName, partitionKey, rowKey, function (err, entity) {
339             log.info('queryEntity results:', err, entity);
340 
341             var routes = JSON.parse(entity.Routes);
342             var oldest = Date.now();
343             var newRoutes = [];
344             for (var index in routes) {
345                 var route = routes[index];
346                 if (route.Name !== routeNameOrToken && route.Token !== routeNameOrToken) {
347                     if (oldest < route.Expiration) {
348                         oldest = route.Expiration;
349                     }
350                     newRoutes.push(route);
351                 }
352             }
353 
354             if (newRoutes.length === 0) {
355                 self.del(userId, deviceId, function (err) {
356                     log.END(err);
357                 });
358             }
359             else {
360                 var registrationEntity = {
361                     'PartitionKey': partitionKey,
362                     'RowKey': rowKey,
363                     'TemplateLanguage': entity.TemplateLanguage,
364                     'TemplateVersion': entity.TemplateVersion,
365                     'Service': entity.Service,
366                     'Expiration': (new Date(oldest)).toISOString(),
367                     'Routes': JSON.stringify(newRoutes)
368                 };
369                 self.store.updateEntity(self.tableName, registrationEntity, function (err) {
370                     log.END(err);
371                 });
372             }
373 	});
374     },

376     /**
377      * Delete all registrations for a given user.
378      *
379      * @param userId
380      *   The user to delete.
381      *
382      * @param callback
383      *   Method to invoke when the deletion is done.
384      *   - err: if not null, definition of the error that took place
385      */
386     delAll: function (userId, callback) {
387         var self = this;
388         var log = self.log.child('deleteAll');
389         var errors = null;
390         var partitionKey = this.makePartitionKey(userId);
391 
392         function mapper(entity, entityCallback) {
393             var rowKey = entity.RowKey;
394             log.debug('deleting registration', rowKey);
395             self._delEntity(
396                 partitionKey, rowKey, function (err) {
397                     if (err) {
398                         if (errors === null) {
399                             errors = {};
400                         }
401                         errors[rowKey] = err;
402                     }
403                     entityCallback(null, null);
404                 });
405         }
406 
407         function reducer(err, results) {
408             callback(errors);
409             log.END();
410         }
411 
412         log.BEGIN(userId);
413 
414         self._getEntities(
415             partitionKey, function (err, registrationEntities) {
416                 if (err || registrationEntities === null || registrationEntities.length === 0) {
417                     callback(err);
418                 }
419                 else {
420                     async.map(registrationEntities, mapper, reducer);
421                 }
422             });
423     }
