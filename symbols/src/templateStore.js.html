1 /**
  2  * @fileOverview Defines the TemplateStore prototype and its methods.
  3  */
  4 module.exports = TemplateStore;
  5 
  6 var azure = require('azure');
  7 var config = require('./config');
  8 var Q = require('q');
  9 
 10 /**
 11  * Initializes a new TemplateStore object. A TemplateStore reads from and writes to an Azure table store.
 12  *
 13  * @class TemplateStore
 14  *
 15  * @param {String} name the name of the table store to use/create
 16  * @param {Function} callback the function to invoke when the table store exits
 17  */
 18 function TemplateStore(tableName, callback) {
 19     var self = this;
 20     var log = this.log = config.log('TemplateStore');
 21     log.BEGIN(tableName);
 22     if (tableName === undefined) {
 23         tableName = config.registrations_table_name;
 24     }
 25 
 26     this.tableName = tableName;
 27     this.store = azure.createTableService();
 28     this.findKeyGenerators = [
 29         this.makeKey.bind(this),
 30         this.makeBaseLanguageKey.bind(this),
 31         this.makeDefaultLanguageKey.bind(this)
 32     ];
 33 
 34     var checkCreateTableIfNotExists = function(err, b, c) {
 35         if (err) {
 36             log.error(err);
 37             if (err.statusCode === 409) {
 38                 Q.delay(1000)   // Wait for one second and try again.
 39                     .then(function () {
 40                         self.store.createTableIfNotExists(tableName, checkCreateTableIfNotExists);
 41                     });
 42                 return;
 43             }
 44         }
 45 
 46         log.END(err, b, c);
 47         if (callback) callback(err);
 48     };
 49 
 50     this.store.createTableIfNotExists(tableName, checkCreateTableIfNotExists);
 51 }
 52 
 53 TemplateStore.prototype = {
 54 
 55     /**
 56      * Find templates that match the given parameters.
 57      */
 58     findTemplates: function(eventId, registrations, callback) {
 59         var self = this;
 60         var log = self.log.child('findTemplates');
 61 
 62         log.BEGIN(eventId, registrations);
 63 
 64         var query = azure.TableQuery.select()
 65                                     .from(this.tableName)
 66                                     .where('PartitionKey eq ?', this.makePartitionKey(eventId.toString()));
 67 
 68         this.store.queryEntities(query, function(err, found) {
 69             var entity, bits, routeName, templateVersion, templateLanguage,
 70                 service, key, matches, routes, routeIndex, route;
 71             if (err) {
 72                 log.error('queryEntities error:', err);
 73                 callback(err, null);
 74                 return;
 75             }
 76             else if (found.length === 0) {
 77                 log.info('queryEntities returned no entities');
 78                 callback(null, found);
 79                 log.END();
 80                 return;
 81             }
 82 
 83             var foundMap = {};
 84             for (var foundIndex in found) {
 85                 log.debug('found:', found);
 86                 entity = found[foundIndex];
 87                 key = self.makeKey(entity.Route, entity.TemplateVersion, entity.TemplateLanguage, entity.Service);
 88                 bits = foundMap[key];
 89                 if (! bits) {
 90                     bits = [];
 91                     foundMap[key] = bits;
 92                 }
 93                 bits.push(entity);
 94             }
 95 
 96             matches = [];
 97             for (var registrationIndex in registrations) {
 98                 var registration = registrations[registrationIndex];
 99                 templateVersion = registration.templateVersion;
100                 templateLanguage = registration.templateLanguage;
101                 service = registration.service;
102                 routes = registration.routes;
103                 for (routeIndex in routes) {
104                     route = routes[routeIndex];
105                     routeName = route.name;
106                     for (var genIndex in self.findKeyGenerators) {
107                         key = self.findKeyGenerators[genIndex](routeName, templateVersion, templateLanguage,
108                                                                service);
109                         log.debug('looking for key:', key);
110                         if (key === null) {
111                             continue;
112                         }
113 
114                         var matched = foundMap[key];
115                         if (matched !== undefined) {
116                             for (var matchIndex in matched) {
117                                 var template = null;
118                                 try {
119                                     template = JSON.parse(matched[matchIndex].Content);
120                                 } catch (err) {
121                                     log.error('failed to parse template:', matched[matchIndex].Content);
122                                 }
123 
124                                 if (template !== null) {
125                                     var match = {
126                                         'service': service,
127                                         'token': route.token,
128                                         'template': JSON.parse(matched[matchIndex].Content)
129                                     };
130                                     log.debug('found match:', match);
131                                     matches.push(match);
132                                 }
133                             }
134                         }
135                     }
136                 }
137             }
138 
139             callback(null, matches);
140             log.END();
141         });
142     },
143 
144     /**
145      * Add a new template to the table store.
146      */
147     addTemplate: function(params, callback) {
148         var self = this;
149         var log = self.log.child('addTemplate');
150         log.BEGIN(params);
151 
152         var partitionKey = self.makePartitionKey(params.eventId.toString());
153         log.debug('partitionKey:', partitionKey);
154 
155         var rowKey = self.makeRowKey(params.notificationId, params.route, params.templateVersion,
156                                      params.templateLanguage, params.service);
157         log.debug('rowKey:', rowKey);
158 
159         var templateEntity = {
160             'PartitionKey': partitionKey,
161             'RowKey': rowKey,
162             'Route': params.route.toString(),
163             'TemplateVersion': params.templateVersion.toString(),
164             'TemplateLanguage': params.templateLanguage.toString(),
165             'Service': params.service.toString(),
166             'Content': JSON.stringify(params.template)
167         };
168 
169         self.store.updateEntity(self.tableName, templateEntity, function(err, tmp) {
170             if (err !== null) {
171                 log.warn('TableService.updateEntity error:', err);
172                 self.store.insertEntity(self.tableName, templateEntity, callback);
173             }
174             else {
175                 callback(err, tmp);
176             }
177             log.END();
178         });
179     },
180 
181     /**
182      * Remove a template from the table store.
183      */
184     removeTemplate: function(params, callback) {
185         var self = this;
186         var log = self.log.child('removeTemplate');
187         log.BEGIN(params);
188 
189         var partitionKey = self.makePartitionKey(params.eventId.toString());
190         log.debug('partitionKey:', partitionKey);
191 
192         var rowKey = self.makeRowKey(params.notificationId, params.route, params.templateVersion,
193                                      params.templateLanguage, params.service);
194         log.debug('rowKey:', rowKey);
195 
196         var templateEntity = {
197             'PartitionKey': partitionKey,
198             'RowKey': rowKey
199         };
200 
201         self.store.deleteEntity(self.tableName, templateEntity, function(err, tmp) {
202             if (err) {
203                 log.error('TableStore.deleteEntity error:', err);
204             }
205             callback(err, tmp);
206         });
207     },
208 
209     /**
210      * Make a template key
211      */
212     makeKey: function(routeName, templateVersion, templateLanguage, service) {
213         return routeName + '_' + templateVersion + '_' + templateLanguage + '_' + service + '_';
214     },
215 
216     /**
217      * Make a base-language template key
218      */
219     makeBaseLanguageKey: function(routeName, templateVersion, templateLanguage, service) {
220         var pos = templateLanguage.indexOf('-');
221         if (pos > 0) {
222             return this.makeKey(routeName, templateVersion, templateLanguage.substr(0, pos), service);
223         }
224         return null;
225     },
226 
227     /**
228      * Make a base-language template key
229      */
230     makeDefaultLanguageKey: function(routeName, templateVersion, templateLanguage, service) {
231         if (templateLanguage.substr(0, 2) !== 'en'){
232             return this.makeKey(routeName, templateVersion, 'en', service);
233         }
234         return null;
235     },
236 
237     /**
238      * Make a table store partition key.
239      */
240     makePartitionKey: function(value) {
241         return new Buffer(value).toString('base64');
242     },
243 
244     /**
245      * Make a table store row key
246      */
247     makeRowKey: function(notificationId, routeName, templateVersion, templateLanguage, service) {
248         return new Buffer(this.makeKey(routeName, templateVersion, templateLanguage, service) +
249                           notificationId.toString()).toString('base64');
250     },
251 
252     /**
253      * Convert a key into its individual components
254      */
255     getKeyBits: function(rowKey) {
256         return rowKey.split('_');
257     }
258 };
