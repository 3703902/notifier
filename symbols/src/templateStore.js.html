1 /**
  2  * @fileOverview Defines the TemplateStore prototype and its methods.
  3  */
  4 module.exports = TemplateStore;
  5 
  6 var azure = require('azure');
  7 var config = require('./config');
  8 var Q = require('q');
  9 var TemplateCache = require('./templateCache');

 11 /**
 12  * Initializes a new TemplateStore object. A TemplateStore reads from and writes to an Azure table store.
 13  *
 14  * @class TemplateStore
 15  *
 16  * @param {String} name the name of the table store to use/create
 17  * @param {Function} callback the function to invoke when the table store exits
 18  */
 19 function TemplateStore(tableName, callback) {
 20     var self = this;
 21     var log = this.log = config.log('TemplateStore');
 22     log.BEGIN(tableName);
 23     if (tableName === undefined) {
 24         tableName = config.registrations_table_name;
 25     }
 26 
 27     self.tableName = tableName;
 28     self.store = azure.createTableService();
 29     self.cache = new TemplateCache();
 30     self.pendingGet = {};
 31 
 32     var checkCreateTableIfNotExists = function(err, b, c) {
 33         if (err) {
 34             log.error(err);
 35             if (err.statusCode === 409) {
 36                 Q.delay(1000)   // Wait for one second and try again.
 37                     .then(function () {
 38                         self.store.createTableIfNotExists(tableName, checkCreateTableIfNotExists);
 39                     });
 40                 return;
 41             }
 42         }
 43 
 44         log.END(err, b, c);
 45         if (callback) callback(err);
 46     };
 47 
 48     self.store.createTableIfNotExists(tableName, checkCreateTableIfNotExists);
 49     log.END();
 50 }

 52 TemplateStore.prototype = {
 53 
 54     /**
 55      * Find templates in the internal cache that match the given parameters.
 56      */
 57     find: function(eventId, registrations, callback) {
 58         var self = this;
 59         var log = self.log.child('find');
 60         log.BEGIN(eventId, registrations);
 61 
 62         var partitionKey = self.makePartitionKey(eventId);
 63         var found = self.cache.get(partitionKey, registrations);
 64         if (typeof found === 'undefined') {
 65             var pending = self.pendingGet[partitionKey];
 66             if (typeof pending !== 'undefined') {
 67                 pending.push([registrations, callback]);
 68                 return;
 69             }
 70 
 71             pending = [];
 72             pending.push([registrations, callback]);
 73             self.pendingGet[partitionKey] = pending;
 74 
 75             self.getEntities(eventId, function (err, found) {
 76                 if (err) {
 77                     log.error('failed to fetch templates:', err);
 78                 }
 79                 else {
 80                     self.cache.set(partitionKey, found);
 81                 }
 82 
 83                 pending = self.pendingGet[partitionKey];
 84                 delete self.pendingGet[partitionKey];
 85 
 86                 // Visit all of the outstanding requests for these templates and invoke their callback.
 87                 for (var index in pending) {
 88                     if (err) {
 89                         pending[index][1](err, null);
 90                     }
 91                     else {
 92                         self.find(eventId, pending[index][0], pending[index][1]);
 93                     }
 94                 }
 95             });
 96             return;
 97         }
 98 
 99         callback(null, found);
100         log.END();
101     },
102 
103     /**
104      * Get all templates associated with a given eventId.
105      *
106      * @param {Integer} eventId the eventId to look for
107      *
108      * @param {Function} callback the function to invoke when the query is done
109      *
110      * @return {Object} results of the query, with templates grouped by their notificationId values
111      */
112     getEntities: function(eventId, callback) {
113         var self = this;
114         var log = self.log.child('getEntities');
115 
116         log.BEGIN(eventId);
117         var partitionKey = self.makePartitionKey(eventId);
118         var query = azure.TableQuery.select()
119                                     .from(self.tableName)
120                                     .where('PartitionKey eq ?', partitionKey);
121         self.store.queryEntities(query, function(err, entities) {
122             var entity, entityIndex;
123             if (err) {
124                 log.error('queryEntities error:', err);
125                 callback(err, null);
126                 return;
127             }
128             callback(null, entities);
129         });
130     },
131 
132     getTemplates: function(params, callback) {
133         var self = this;
134         var log = self.log.child('getTemplates');
135 
136         log.BEGIN(params);
137         var partitionKey = self.makePartitionKey(params.eventId);
138         log.debug('partitionKey:', partitionKey);
139 
140         var query = azure.TableQuery.select()
141                                     .from(self.tableName)
142                                     .where('PartitionKey eq ?', partitionKey);
143         self.store.queryEntities(query, function(err, entities) {
144             var found, entity, entityIndex;
145             if (err) {
146                 log.error('queryEntities error:', err);
147                 callback(err, null);
148                 return;
149             }
150 
151             callback(null, self.entitiesToJson(params.eventId, entities));
152         });
153     },
154 
155     entitiesToJson: function(eventId, entities) {
156         var converted = [];
157         var entity, entityIndex;
158         for (entityIndex in entities) {
159             entity = entities[entityIndex];
160             converted.push(
161                 {
162                     eventId: parseInt(eventId),
163                     notificationId: parseInt(entity.NotificationId),
164                     route: entity.Route,
165                     templateVersion: entity.TemplateVersion,
166                     templateLanguage: entity.TemplateLanguage,
167                     service: entity.Service,
168                     template: JSON.parse(entity.Content)
169                 });
170         }
171         return converted;
172     },
173 
174     /**
175      * Get a template from the table store.
176      */
177     get: function(params, callback) {
178         var self = this;
179         var log = self.log.child('get');
180         log.BEGIN(params);
181 
182         var partitionKey = self.makePartitionKey(params.eventId);
183         log.debug('partitionKey:', partitionKey);
184 
185         var rowKey = self.makeRowKey(params.notificationId, params.route, params.templateVersion,
186                                      params.templateLanguage, params.service);
187         log.debug('rowKey:', rowKey);
188 
189         var query = azure.TableQuery.select()
190                                     .from(self.tableName)
191                                     .where('PartitionKey eq ? and RowKey eq ?', partitionKey, rowKey);
192         self.store.queryEntities(query, function(err, entities) {
193             if (err) {
194                 log.error('queryEntities error:', err);
195                 callback(err, null);
196                 return;
197             }
198 
199             callback(null, self.entitiesToJson(params.eventId, entities));
200         });
201     },
202 
203     /**
204      * Add a new template to the table store.
205      */
206     add: function(params, callback) {
207         var self = this;
208         var log = self.log.child('add');
209         log.BEGIN(params);
210 
211         var partitionKey = self.makePartitionKey(params.eventId);
212         log.debug('partitionKey:', partitionKey);
213 
214         var rowKey = self.makeRowKey(params.notificationId, params.route, params.templateVersion,
215                                      params.templateLanguage, params.service);
216         log.debug('rowKey:', rowKey);
217 
218         var entity = {
219             'PartitionKey': partitionKey,
220             'RowKey': rowKey,
221             'NotificationId': params.notificationId.toString(),
222             'Route': params.route.toString(),
223             'TemplateVersion': params.templateVersion.toString(),
224             'TemplateLanguage': params.templateLanguage.toString(),
225             'Service': params.service.toString(),
226             'Content': JSON.stringify(params.template)
227         };
228 
229         log.debug('entity:', entity);
230 
231         self.store.updateEntity(self.tableName, entity, function(err, tmp) {
232             log.debug('updateEntity err:', err, 'tmp:', tmp);
233             if (err !== null) {
234                 log.debug('attempting to insert:', entity);
235                 self.store.insertEntity(self.tableName, entity, function (err, tmp) {
236                     log.debug('insertEntity err: ', err, ' tmp: ', tmp);
237                     if (err === null) {
238                         self.cache.add(entity);
239                     }
240                     callback(err, tmp);
241                     log.END();
242                 });
243             }
244             else {
245                 self.cache.add(entity);
246                 callback(err, tmp);
247                 log.END();
248             }
249         });
250     },
251 
252     /**
253      * Remove a template from the table store.
254      *
255      * @param {Object} params specification of the template to remove
256      * @param {function(err,entity)} callback function to invoke upon completion of the deletion
257      */
258     del: function(params, callback) {
259         var self = this;
260         var log = self.log.child('del');
261         log.BEGIN(params);
262 
263         var partitionKey = self.makePartitionKey(params.eventId);
264         log.debug('partitionKey:', partitionKey);
265 
266         var rowKey = self.makeRowKey(params.notificationId, params.route, params.templateVersion,
267                                      params.templateLanguage, params.service);
268         log.debug('rowKey:', rowKey);
269 
270         var entity = {
271             'PartitionKey': partitionKey,
272             'RowKey': rowKey,
273             'NotificationId': params.notificationId.toString(),
274             'Route': params.route.toString(),
275             'TemplateVersion': params.templateVersion.toString(),
276             'TemplateLanguage': params.templateLanguage.toString(),
277             'Service': params.service.toString()
278         };
279 
280         log.debug(entity);
281 
282         self.cache.del(entity);
283 
284         self.store.deleteEntity(self.tableName, entity, function(err, tmp) {
285             if (err) {
286                 log.error('TableStore.deleteEntity error:', err);
287             }
288             callback(err, tmp);
289         });
290     },

292     /**
293      * Make a table store partition key. For templates, the partition key is the notification event ID. However, we
294      * encode in base64 just to be safe against values harmful to Azure table store.
295      *
296      * @param {String|Number} value the value to use for the key
297      * @return {String} value safe to use in Azure table store for the partition key.
298      */
299     makePartitionKey: function(value) {
300         return (new Buffer(value.toString())).toString('base64');
301     },
302 
303     /**
304      * Make a table store row key. This will point to a specific template payload definition. As such, it must be
305      * uniquely specified in 5 dimensions: notification ID, route name, template version, template language, and
306      * notification delivery service.
307      *
308      * @param {String|Integer} notificationId the notification ID associated with the template
309      * @param {String} routeName the route name associated with the template
310      * @param {String} templateVersion the version of the template being keyed
311      * @param {String} templateLanguage the language of the template being keyed
312      * @param {String} service the notification delivery service to use
313      * @return {String} concatenated value of the above encoded in base64 for Azure table store safety
314      */
315     makeRowKey: function(notificationId, routeName, templateVersion, templateLanguage, service) {
316         return (new Buffer(routeName + '_' + templateVersion + '_' + templateLanguage + '_' + service + '_' +
317                            notificationId)).toString('base64');
318     }
