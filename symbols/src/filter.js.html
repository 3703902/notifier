1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines a Filter prototype and its methods.
  5  */
  6 module.exports = Filter;
  7 
  8 /**
  9  * Filter constructor.
 10  *
 11  * @param {Object} init defines the attributes to filter against
 12  *
 13  * @class
 14  *
 15  * A Filter object determines of a device registration is to be used for notifications. A Filter contains one or more
 16  * attributes names that correspond to attribute names found in a registration or registration route. Each attribute
 17  * name is associated with an array of allowed values. If a registration/route pair given to the passed method does not
 18  * have the required values, the registration/route will fail to pass the filter.
 19  */
 20 function Filter(init) {
 21     this.log = require('./config').log('Filter');
 22     this.log.BEGIN(init);
 23     if (init === undefined || init === {}) {
 24         init = null;
 25     }
 26     this.filters = init;
 27 }
 28 
 29 /**
 30  * Filter prototype.
 31  *
 32  * Defines the methods available to a Filter instance.
 33  */
 34 Filter.prototype = {
 35 
 36     /**
 37      * Apply the filter to a registration and route defintion.
 38      *
 39      * @param {Object} registration a registration definition. See {@link Registrar#RegistrationModel} for the
 40      *                              definition.
 41      * @param {Object} route a registration route definition. See {@link Registrar#RouteModel} for the definition.
 42      * @return true if passes the filter, false otherwise.
 43      */
 44     passed: function(registration, route) {
 45         var log = this.log.child('passed');
 46         log.BEGIN(registration, route);
 47         if (this.definition === null) {
 48             log.END('true');
 49             return true;
 50         }
 51 
 52         for (var key in this.filters) {
 53             var validValues = this.filters[key];
 54             var value = registration[key];
 55             if (typeof value !== 'undefined') {
 56                 if (validValues.indexOf(value) === -1) {
 57                     log.END('false');
 58                     return false;
 59                 }
 60             }
 61 
 62             value = route[key];
 63             if (typeof value !== 'undefined') {
 64                 if (validValues.indexOf(value) === -1) {
 65                     log.END('false');
 66                     return false;
 67                 }
 68             }
 69         }
 70 
 71         log.END('true');
 72         return true;
 73     }
 74 }
