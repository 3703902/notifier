1 "use strict";
  2 
  3 /**
  4  * @fileOverview Defines the configuration parameters for the service.
  5  */
  6 module.exports = undefined;
  7 
  8 // Load in sensitive parameters
  9 var fs = require('fs');
 10 var priv = require('./private/config');
 11 var LoggerUtils = require('./loggerUtils');
 12 
 13 /**
 14  * Configuration module.
 15  *
 16  * @class config
 17  *
 18  * Configuration parameters for the notifier service.
 19  */
 20 function Config () {
 21 
 22     var log4js = require('log4js');
 23 
 24     /**
 25      * Logging configuration for the server.
 26      */
 27     log4js.configure(
 28         {
 29             // Define the log appenders to use. These apply to all loggers.
 30             "appenders": [
 31                 {
 32                     "type": "file",
 33                     "filename": "notifier.log",
 34                     "layout": { "type": "basic" }
 35                 },
 36                 {
 37                     "type": "console",
 38                     "layout": { "type": "basic" }
 39                 }
 40             ],
 41 
 42             // Define the levels to accept for messages from a logger.
 43             "levels": {
 44                 "[default]":  "ERROR"
 45             }
 46         }
 47     );
 48 
 49     /**
 50      * Determines if SSL authentication is enabled.
 51      */
 52     this.ssl_authentication_enabled = false;
 53 
 54     /**
 55      * Determines if unauthenticated traffic will be accepted.
 56      */
 57     this.ssl_reject_unauthorized = false;
 58 
 59     /**
 60      * The contents of the server key for SSL authentication.
 61      */
 62     this.ssl_server_key = fs.readFileSync('certs/server.key');
 63 
 64     /**
 65      * The passphrase for the above server key.
 66      */
 67     this.ssl_server_key_passphrase = '';
 68 
 69     /**
 70      * The contents of the server certificate file for SSL authentication.
 71      */
 72     this.ssl_server_certificate = fs.readFileSync('certs/server.cert');
 73 
 74     /**
 75      * The contents of the certificate authority certificate file for SSL authentication.
 76      */
 77     this.ssl_ca_certificate = fs.readFileSync('certs/ca.cert');
 78 
 79     /**
 80      * Shortcut to log4js.getLogger procedure. Allows other modules to do ```config.log('moduleName')```
 81      * @type {Function}
 82      */
 83     this.log = log4js.getLogger;
 84 
 85     /**
 86      * Path to local file that hold's the last fetched access_token. This will be read at startup if it exists.
 87      * @type {String}
 88      */
 89     this.wns_access_token_cache = 'wns_access_token_cache.js',
 90 
 91     /**
 92      * FQDN of the host to use to deliver WNS notification payloads.
 93      * @type {String}
 94      */
 95     this.wns_host = 'cloud.notify.windows.com';
 96 
 97     /**
 98      * The client secret associated with the Windows application
 99      * @type {String}
100      */
101     this.wns_client_secret = priv.wns_client_secret;
102 
103     /**
104      * The package name associated with the Windows application
105      * @type {String}
106      */
107     this.wns_package_name = priv.wns_package_name;
108 
109     /**
110      * The package SID associated with the Windows application
111      * @type {String}
112      */
113     this.wns_package_sid = priv.wns_package_sid;
114 
115     /**
116      * The URL to use to fetch new WNS access tokens for the above application credentials.
117      * @type {String}
118      */
119     this.wns_access_token_url = 'https://login.live.com/accesstoken.srf';
120 
121     /**
122      * The FQDN of the host to use to deliver APNs notifcation payloads.
123      * @type {String}
124      */
125     this.apns_service_host = 'gateway.sandbox.push.apple.com'; // or 'gateway.push.apple.com'
126 
127     /**
128      * The port of the host to connect to for APNs notifications.
129      * @type {Number}
130      */
131     this.apns_service_port = 2195;
132 
133     /**
134      * The FQDN of the host to connect to for APNs feedback.
135      * @type {String}
136      */
137     this.apns_feedback_host = 'feedback.sandbox.push.apple.com'; // or 'feedback.push.apple.com'
138 
139     /**
140      * The port of the host to connect to for APNs feedback.
141      * @type {Number}
142      */
143     this.apns_feedback_port = 2196;
144 
145     /**
146      * The number of seconds between checks for APNs feedback (zero to disable).
147      * @type {Number}
148      */
149     this.apns_feedback_interval = 3600;
150 
151     /**
152      * The max number of entries in the APNs device cache used to locate a user when Apple's feedback service says the
153      * device token is no longer valid.
154      */
155     this.apns_device_cache_size = 10000;
156 
157     /**
158      * The file holding the root certificate used to authenticate the remote server.
159      * @type {String}
160      */
161     this.apns_root_certificate_file = 'private/EntrustRootCertificationAuthority.pem';
162 
163     /**
164      * The file holding the private key certificate used to authenticate to the remote server.
165      * @type {String}
166      */
167     this.apns_client_private_key_file = 'private/NotificationHubTestDevPush_pkey.pem';
168 
169     /**
170      * The file holding the credentials used to authenticate to the remote server for application notifications.
171      * @type {String}
172      */
173     this.apns_client_certificate_file = 'private/NotificationHubTestDevPush_cred.pem';
174 
175     /**
176      * The Azure storage account to use for table stores.
177      * @type {String}
178      */
179     this.azure_storage_account = priv.azure_storage_account;
180 
181     /**
182      * The access key for the above Azure storage account
183      * @type {String}
184      */
185     this.azure_storage_access_key = priv.azure_storage_access_key;
186 
187     // Make sure our environment reflects these Azure settings.
188     process.env.AZURE_STORAGE_ACCOUNT = this.azure_storage_account;
189     process.env.AZURE_STORAGE_ACCESS_KEY = this.azure_storage_access_key;
190 
191     /**
192      * The authorization key from Google for sending GCM messages
193      * @type {String}
194      */
195     this.gcm_authorization_key = priv.gcm_authorization_key;
196 
197     /**
198      * The URI for the GCM server
199      * @type {String}
200      */
201     this.gcm_uri = 'https://android.googleapis.com/gcm/send';
202 
203     /**
204      * Name of the Azure table to create and use for user registrations.
205      * @type {String}
206      */
207     this.registrations_table_name = 'notifierRegistrations';
208 
209     /**
210      * Name of the Azure table to create and use for notification templates.
211      * @type {String}
212      */
213     this.templates_table_name = 'notifierTemplates';
214 
215     /**
216      * The regular-expression (RE) that matches placeholders in a template.
217      *
218      * Placeholders are signaled by two '@' characters, followed by a name of alphanumeric characters and underscores.
219      * There may also be an optional default value, signaled by a '=' character after the name. The stock one below
220      * matches fields such as: @@ABC@@, @@FooBar=default value@@
221      *
222      * NOTE: if you change this, odds are you have to change the RE processing code in payloadGenerator.js
223      *
224      * @type {String}
225      */
226     this.templates_placeholder_re = '@@([A-Za-z0-9_]+)(=([^@]*))?@@';
227 }
228 
229 module.exports = new Config();
