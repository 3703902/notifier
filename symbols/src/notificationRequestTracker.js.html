1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the NotificationRequestTracker prototype and its methods.
  5  */
  6 module.exports = NotificationRequestTracker;
  7 
  8 var Deque = require('./deque');
  9 var config = require('./config');
 10 
 11 /**
 12  * NotificationRequestTracker constructor
 13  *
 14  * @class
 15  *
 16  * A NotificationRequestTracker instance maintains a LIFO size-limited queue that contains Node instances that define a
 17  * relationship between a user ID and an APNs device token.
 18  */
 19 function NotificationRequestTracker(maxSize) {
 20     this.log = config.log('NotificationRequestTracker');
 21     this.maxSize = maxSize;
 22     this.active = new Deque();
 23     this.mapping = {};
 24 }
 25 
 26 NotificationRequestTracker.prototype = {
 27 
 28     /**
 29      * Create a new Node instance and add to the LIFO size-limited queue.
 30      *
 31      * @param {NotificationRequest} request the request to remember
 32      */
 33     add: function (request) {
 34         var log = this.log.child('add');
 35         log.BEGIN(request);
 36 
 37         var node = this.mapping[request.token];
 38         if (typeof node !== 'undefined') {
 39 
 40             // Node already exist for this token. Reuse by removing from the queue.
 41             node.extricate();
 42             node.data = request;
 43         }
 44         else {
 45             if (this.active.size() === this.maxSize) {
 46 
 47                 // Queue is full of past requests. Reuse the oldest one.
 48                 node = this.active.popBackNode();
 49                 delete this.mapping[node.data.token];
 50                 node.data = request;
 51             }
 52             else {
 53 
 54                 // Space availabe in queue. Allocate new node.
 55                 node = new Deque.Node(request);
 56             }
 57             this.mapping[request.token] = node;
 58         }
 59 
 60         this.active.pushFrontNode(node);
 61         log.END();
 62     },
 63 
 64     /**
 65      * Attempt to locate the NoificationRequest that belongs to a given iOS device token.
 66      *
 67      * @param {String} token the route token to look for
 68      * @return {NotificationRequest} related request or null.
 69      */
 70     find: function(token) {
 71         var log = this.log.child('find');
 72         log.BEGIN(token);
 73         var node = this.mapping[token];
 74         var request = node ? node.data: null;
 75         if (request) {
 76             this.active.remove(node);
 77             delete this.mapping[token];
 78         }
 79 
 80         log.END(request);
 81         return request;
 82     }
 83 };
