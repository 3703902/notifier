1 /**
  2  * @fileOverview Defines the TemplateCache prototype and its methods.
  3  */
  4 module.exports = TemplateCache;
  5 
  6 var config = require('./config');
  7 var TemplateParser = require('./templateParser');
  8 var NotificationRequest = require('./notificationRequest');
  9 
 10 /**
 11  * TemplateCache constructor.
 12  *
 13  * @class
 14  *
 15  * A TemplateCache records in-memory templates fetched from an Azure table store. It consists of a top-level hash, with
 16  * keys made up of Azure table store partition keys that are derived from notification event ID values. Each entry in
 17  * this hash is itself a hash, where the keys are made up of various template attributes, and the value is an array
 18  * (there can be more than one template associated with a notification event ID).
 19  */
 20 function TemplateCache() {
 21     var log = this.log = config.log('TemplateCache');
 22     log.BEGIN();
 23 
 24     this.cache = {};
 25 
 26     /**
 27      * Holds the functions to invoke to generate search keys. For a registration with a templateLanguage value of
 28      * "de-AU" (Austrian German), there are three queries that can be made:
 29      *
 30      * - de-AU - original language value
 31      * - de - base language value
 32      * - en - default language value
 33      *
 34      * These functions are invoked in sequence, and if they return a non-null value, then the key they generated will
 35      * be used to search for a template.
 36      *
 37      * @type function array
 38      */
 39     this.findKeyGenerators = [
 40         this.makeKey,
 41         this.makeBaseLanguageKey,
 42         this.makeDefaultLanguageKey
 43     ];
 44     log.END();
 45 }
 46 
 47 /**
 48  * TemplateCache.Entry constructor.
 49  *
 50  * @class
 51  *
 52  * Represents entries in the cache. Each one has a parsed template function to be used to generate notification
 53  * payloads.
 54  *
 55  * @param {Object} entity the Azure table store entity to remember. Records the entity.NotificationId and the
 56  * entity.Content values. The latter is then parsed into a function that can be used to generate notification payloads.
 57  */
 58 TemplateCache.Entry = function(entity) {
 59     var log = config.log('TemplateCache.Entry');
 60     log.BEGIN(entity);
 61     this.notificationId = entity.NotificationId;
 62     log.debug('notificationId:', this.notificationId);
 63 
 64     try {
 65         log.debug('attempting to convert entity.Content into JSON:', entity.Content);
 66         this.template = JSON.parse(entity.Content);
 67         log.debug('done:', this.template);
 68         try {
 69             log.debug('attempting to parse template.content:', this.template.content);
 70             this.template.content = TemplateParser.parse(this.template.content);
 71             log.debug('done:', this.template.content);
 72         }
 73         catch(err) {
 74             log.error('failed to parse template:', this.template.content);
 75             log.error(err);
 76             this.template.content = null;
 77         }
 78     }
 79     catch (err) {
 80         log.error('failed to parse entity.Content as JSON:', entity.Content);
 81         log.error(err);
 82         this.content = null;
 83     }
 84 
 85     log.END();
 86 };
 87 
 88 TemplateCache.prototype = {
 89 
 90     /**
 91      * Make a template key.
 92      *
 93      * @param {String} routeName the name of the route associated with the template.
 94      * @param {String} templateVersion the version of the template
 95      * @param {String} templateLanguage the language of the template
 96      * @param {String} service the name of the OS-specific service used for notification delivery
 97      * @return {String} key made up of the given values.
 98      *
 99      * @private
100      */
101     makeKey: function(routeName, templateVersion, templateLanguage, service) {
102         return routeName + '_' + templateVersion + '_' + templateLanguage + '_' + service + '_';
103     },
104 
105     /**
106      * Make a base-language template key, if one exists. Strips off any '-*' suffix from the templateLanguage value and
107      * returns the value from {makeKey} with that value.
108      *
109      * @param {String} routeName the name of the route associated with the template.
110      * @param {String} templateVersion the version of the template
111      * @param {String} templateLanguage the language of the template
112      * @param {String} service the name of the OS-specific service used for notification delivery
113      * @return {String} key made up of the given values, or null if there was no extension present
114      *
115      * @private
116      */
117     makeBaseLanguageKey: function(routeName, templateVersion, templateLanguage, service) {
118         var pos = templateLanguage.indexOf('-');
119         if (pos > 0) {
120             return this.makeKey(routeName, templateVersion, templateLanguage.substr(0, pos), service);
121         }
122         return null;
123     },
124 
125     /**
126      * Make a default language template key if it would differ from the current templateLanguage base.
127      *
128      * @param {String} routeName the name of the route associated with the template.
129      * @param {String} templateVersion the version of the template
130      * @param {String} templateLanguage the language of the template
131      * @param {String} service the name of the OS-specific service used for notification delivery
132      * @return {String} key made up of the given values, or null if not relevant
133      *
134      * @private
135      */
136     makeDefaultLanguageKey: function(routeName, templateVersion, templateLanguage, service) {
137         if (templateLanguage.substr(0, 2) !== 'en'){
138             return this.makeKey(routeName, templateVersion, 'en', service);
139         }
140         return null;
141     },
142 
143     /**
144      * Fetch templates generators from the cache that match the given partitionKey and user registration settings.
145      * Generates a key template key for each registration route in each registration, and looks for it within the
146      * cache. If found, it creates a new NotificationRequest object based on the cache entry.
147      *
148      * @param {String} partitionKey the key associated with the notification event to use for fetching
149      * @param {Array} registrations  the set of user registrations to use for matching
150      * @return array of {@link NotificationRequest} instances appropriate for the registrations
151      */
152     get: function(partitionKey, registrations) {
153         var log = this.log.child('get');
154         log.BEGIN(partitionKey, registrations);
155 
156         var templates = this.cache[partitionKey];
157         if (typeof templates === 'undefined') {
158             log.END('no cached entry');
159             return templates;
160         }
161 
162         var matches = [];
163         for (var registrationIndex in registrations) {
164             var registration = registrations[registrationIndex];
165             var templateVersion = registration.templateVersion;
166             var templateLanguage = registration.templateLanguage;
167             var service = registration.service;
168             var routes = registration.routes;
169             for (var routeIndex in routes) {
170                 var route = routes[routeIndex];
171                 var routeName = route.name;
172                 for (var keyIndex in this.findKeyGenerators) {
173                     key = this.findKeyGenerators[keyIndex].call(this, routeName, templateVersion, templateLanguage,
174                                                                 service);
175                     if (key === null) {
176                         continue;
177                     }
178 
179                     log.debug('looking for key:', key);
180                     var found = templates[key];
181                     if (typeof found !== 'undefined' && found.length > 0) {
182                         log.debug('found: ', found);
183                         found.forEach(function (entry) {
184                             matches.push(new NotificationRequest(registration.registrationId, service, route.token,
185                                                                  entry.template));
186                         });
187                         break;
188                     }
189                 }
190             }
191         }
192 
193         log.END(matches);
194         return matches;
195     },
196 
197     /**
198      * Replace any existing cache entries with the ones given.
199      *
200      * @param {String} partitionKey the key associated with the notification event to use for fetching
201      * @param {Array} entities  the set of template entities to store in the cache
202      */
203     set: function(partitionKey, entities) {
204         var self = this;
205         var log = this.log.child('set');
206         log.BEGIN(partitionKey, entities);
207 
208         // Clear the cache for this event ID
209         this.cache[partitionKey] = {};
210 
211         // Add all of the entities to the cache.
212         entities.forEach(function (item) {self.add(item);});
213 
214         log.END(this.cache);
215     },
216 
217     /**
218      * Add or replace one template entity in the cache.
219      *
220      * @param {Object} entity the Azure table store entity to remember
221      */
222     add: function(entity)  {
223         var log = this.log.child('add');
224         log.BEGIN(entity);
225 
226         var key = entity.PartitionKey;
227         var templateMap = this.cache[key];
228         if (typeof templateMap === 'undefined') {
229             log.END('no cached entry');
230             return;
231         }
232 
233         var entry = new TemplateCache.Entry(entity);
234         log.debug('entry:', entry);
235 
236         key = this.makeKey(entity.Route, entity.TemplateVersion, entity.TemplateLanguage, entity.Service);
237         log.debug('key:', key);
238 
239         var found = templateMap[key];
240         if (typeof found  === 'undefined') {
241             log.debug('new entry');
242             templateMap[key] = found = [entry];
243         }
244         else {
245             for (var foundIndex in found) {
246                 if (found[foundIndex].notificationId === entity.NotificationId) {
247                     log.debug('replacing entry');
248                     found[foundIndex] = entry;
249                     entry = null;
250                     break;
251                 }
252             }
253 
254             if (entry) {
255                 log.debug('new entry');
256                 found.push(entry);
257             }
258         }
259 
260         log.END(found);
261     },
262 
263     /**
264      * Remove a template entity from the cache.
265      *
266      * @param {Object} entity the Azure table store entity to forget
267      */
268     del: function(entity) {
269         var log = this.log.child('del');
270         log.BEGIN(entity);
271 
272         // Get the template hash in the top-level hash.
273         var key = entity.PartitionKey;
274         var entities = this.cache[key];
275         if (typeof entities === 'undefined') {
276             log.END('no cached entry');
277             return;
278         }
279 
280         key = this.makeKey(entity.Route, entity.TemplateVersion, entity.TemplateLanguage, entity.Service);
281         log.debug('key:', key);
282 
283         // Get the bucket from the template hash
284         var found = entities[key];
285         if (typeof found  === 'undefined') {
286             log.END('not found');
287             return;
288         }
289 
290         // Now look for the matching notificationId within the bucket and remove if found
291         for (var foundIndex in found) {
292             if (found[foundIndex].notificationId === entity.NotificationId) {
293                 log.debug('deleting index:', foundIndex);
294                 found.splice(foundIndex, 1);
295                 break;
296             }
297         }
298 
299         log.END();
300     }
301 };
