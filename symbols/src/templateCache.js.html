1 /**
  2  * @fileOverview Defines the TemplateCache prototype and its methods.
  3  */
  4 module.exports = TemplateCache;
  5 
  6 var config = require('./config');
  7 var TemplateParser = require('./templateParser');
  8 var NotificationRequest = require('./notificationRequest');
  9 
 10 /**
 11  * TemplateCache constructor.
 12  *
 13  * @class
 14  *
 15  * A TemplateCache records in-memory templates fetched from an Azure table store.
 16  */
 17 function TemplateCache() {
 18     var log = this.log = config.log('TemplateCache');
 19     log.BEGIN();
 20     this.cache = {};
 21     this.findKeyGenerators = [
 22         this.makeKey.bind(this),
 23         this.makeBaseLanguageKey.bind(this),
 24         this.makeDefaultLanguageKey.bind(this)
 25     ];
 26     log.END();
 27 }
 28 
 29 /**
 30  * TemplateCache.Entry constructor.
 31  *
 32  * @class
 33  *
 34  * Represents entries in the cache. Each one has a parsed template function to be used to generate notification
 35  * payloads.
 36  */
 37 TemplateCache.Entry = function(entity) {
 38     var log = config.log('TemplateCache.Entry');
 39     log.BEGIN(entity);
 40     this.notificationId = entity.NotificationId;
 41     log.debug('notificationId:', this.notificationId);
 42 
 43     try {
 44         log.debug('attempting to convert entity.Content into JSON:', entity.Content);
 45         this.template = JSON.parse(entity.Content);
 46         log.debug('done:', this.template);
 47         try {
 48             log.debug('attempting to parse template.content:', this.template.content);
 49             this.template.content = TemplateParser.parse(this.template.content);
 50             log.debug('done:', this.template.content);
 51         }
 52         catch(err) {
 53             log.error('failed to parse template:', this.template.content);
 54             log.error(err);
 55             this.template.content = null;
 56         }
 57     }
 58     catch (err) {
 59         log.error('failed to parse entity.Content as JSON:', entity.Content);
 60         log.error(err);
 61         this.content = null;
 62     }
 63 
 64     log.END();
 65 };
 66 
 67 TemplateCache.prototype = {
 68 
 69     /**
 70      * Make a template key
 71      */
 72     makeKey: function(routeName, templateVersion, templateLanguage, service) {
 73         return routeName + '_' + templateVersion + '_' + templateLanguage + '_' + service + '_';
 74     },
 75 
 76     /**
 77      * Make a base-language template key
 78      */
 79     makeBaseLanguageKey: function(routeName, templateVersion, templateLanguage, service) {
 80         var pos = templateLanguage.indexOf('-');
 81         if (pos > 0) {
 82             return this.makeKey(routeName, templateVersion, templateLanguage.substr(0, pos), service);
 83         }
 84         return null;
 85     },
 86 
 87     /**
 88      * Make a default-language template key
 89      */
 90     makeDefaultLanguageKey: function(routeName, templateVersion, templateLanguage, service) {
 91         if (templateLanguage.substr(0, 2) !== 'en'){
 92             return this.makeKey(routeName, templateVersion, 'en', service);
 93         }
 94         return null;
 95     },
 96 
 97     /**
 98      * Fetch the templates from the cache that match the given eventId and user registration settings.
 99      */
100     get: function(eventId, registrations) {
101         var log = this.log.child('get');
102         log.BEGIN(eventId, registrations);
103 
104         var key = eventId.toString();
105         var templates = this.cache[key];
106         if (typeof templates === 'undefined') {
107             log.END('no cached entry');
108             return templates;
109         }
110 
111         var matches = [];
112         for (var registrationIndex in registrations) {
113             var registration = registrations[registrationIndex];
114             var templateVersion = registration.templateVersion;
115             var templateLanguage = registration.templateLanguage;
116             var service = registration.service;
117             var routes = registration.routes;
118             for (var routeIndex in routes) {
119                 var route = routes[routeIndex];
120                 var routeName = route.name;
121                 for (var keyIndex in this.findKeyGenerators) {
122                     key = this.findKeyGenerators[keyIndex](routeName, templateVersion, templateLanguage, service);
123                     if (key === null) {
124                         continue;
125                     }
126 
127                     log.debug('looking for key:', key);
128                     var found = templates[key];
129                     if (typeof found !== 'undefined' && found.length > 0) {
130                         log.debug('found: ', found);
131                         found.forEach(function (entry) {
132                             matches.push(new NotificationRequest(registration.registrationId, service, route.token,
133                                                                  entry.template));
134                         });
135                         break;
136                     }
137                 }
138             }
139         }
140 
141         log.END(matches);
142         return matches;
143     },
144 
145     /**
146      * Add a template entity to the cache.
147      */
148     set: function(eventId, templates) {
149         var log = this.log.child('set');
150         log.BEGIN(eventId, templates);
151 
152         var key = eventId.toString();
153         var templateMap = {};
154         this.cache[key] = templateMap;
155 
156         for (var index in templates) {
157             this.add(eventId, templates[index]);
158         }
159 
160         log.END(this.cache);
161     },
162 
163     add: function(eventId, entity)  {
164         var log = this.log.child('add');
165         log.BEGIN(eventId, entity);
166 
167         var key = eventId.toString();
168         var templateMap = this.cache[key];
169         if (typeof templateMap === 'undefined') {
170             log.END('no cached entry');
171             return;
172         }
173 
174         var entry = new TemplateCache.Entry(entity);
175         log.debug('entry:', entry);
176 
177         key = this.makeKey(entity.Route, entity.TemplateVersion, entity.TemplateLanguage, entity.Service);
178         log.debug('key:', key);
179 
180         var found = templateMap[key];
181         if (typeof found  === 'undefined') {
182             log.debug('new entry');
183             templateMap[key] = found = [entry];
184         }
185         else {
186             for (var foundIndex in found) {
187                 if (found[foundIndex].notificationId === entity.NotificationId) {
188                     log.debug('replacing entry');
189                     found[foundIndex] = entry;
190                     entry = null;
191                     break;
192                 }
193             }
194 
195             if (entry) {
196                 log.debug('new entry');
197                 found.push(entry);
198             }
199         }
200 
201         log.END(found);
202     },
203 
204     del: function(eventId, template) {
205         var log = this.log.child('del');
206         log.BEGIN(eventId, template);
207 
208         var key = eventId.toString();
209         var templates = this.cache[key];
210         if (typeof templates === 'undefined') {
211             log.END('no cached entry');
212             return;
213         }
214 
215         key = this.makeKey(template.Route, template.TemplateVersion, template.TemplateLanguage, template.Service);
216         log.debug('key:', key);
217 
218         var found = templates[key];
219         if (typeof found  === 'undefined') {
220             log.END('not found');
221             return;
222         }
223 
224         for (var foundIndex in found) {
225             if (found[foundIndex].notificationId === template.NotificationId) {
226                 log.debug('deleting index:', foundIndex);
227                 found.splice(foundIndex, 1);
228                 break;
229             }
230         }
231 
232         log.END();
233     },
234 
235     clear: function(eventId) {
236         var log = this.log.child('clear');
237         log.BEGIN(eventId);
238         var key = eventId.toString();
239         delete this.cache[key];
240         log.END();
241     }
242 };
