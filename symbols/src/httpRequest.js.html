1 'use strict';
  2 
  3 module.exports = HTTPRequest;
  4 
  5 var config = require('./config');
  6 var Q = require('q');
  7 var request = require('request');
  8 var mlog = config.log('HTTPRequest');
  9 
 10 function HTTPRequest(req, callback) {
 11     mlog.BEGIN(JSON.stringify(req));
 12 
 13     if (typeof req.retryTimeout === 'undefined') {
 14         req.retryTimeout = 1 * 1000;
 15     }
 16 
 17     if (typeof req.retryStatusCheck === 'undefined') {
 18         req.retryStatusCheck =
 19             function (statusCode) {
 20                 return statusCode >= 500;
 21             };
 22     }
 23 
 24     if (typeof req.secondsToLive === 'undefined') {
 25         req.secondsToLive = 30;
 26     }
 27 
 28     var lifetime = 0;
 29     if (req.secondsToLive > 0) {
 30         lifetime = Date.now() + req.secondsToLive * 1000;
 31     }
 32 
 33     return initiateRequest(req, lifetime, callback);
 34 }
 35 
 36 function initiateRequest(req, lifetime, callback) {
 37     var log = mlog.child('initiateRequest');
 38     log.BEGIN(JSON.stringify(req), lifetime);
 39 
 40     if (lifetime > 0 && lifetime <= Date.now()) {
 41         var err = new Error('HTTPRequest: failed to process ' + JSON.stringify(req));
 42         log.error(err);
 43         callback(err, null);
 44         return err;
 45     }
 46 
 47     // Invoke request() method wrapping it in a Q promise.
 48     return Q.ncall(request, null, req)
 49         .then(function (args) {
 50                   var resp = args[0];
 51                   var body = args[1];
 52                   // Success - but is it something we want?
 53                   if (req.retryStatusCheck(resp.statusCode)) {
 54                       log.warn('failed response code check:', resp.statusCode);
 55                       log.debug('retrying in ',req.retryTimeout, 'ms');
 56                       return Q.delay(req.retryTimeout)
 57                           .then(function () {
 58                                     return initiateRequest(req, lifetime, callback);
 59                                 });
 60                   }
 61                   log.info('response:', resp.statusCode);
 62                   log.debug('body:', body);
 63                   return callback(null, resp, body);
 64               },
 65               function (err) {
 66                   // Wait configured amount of time and redo the request
 67                   log.error('error:', err);
 68                   log.debug('retrying in ',req.retryTimeout, 'ms');
 69                   return Q.delay(req.retryTimeout)
 70                       .then(function () {
 71                                 return initiateRequest(req, lifetime, callback);
 72                             });
 73             });
 74 }
