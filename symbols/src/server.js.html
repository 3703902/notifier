1 'use strict';
  2 
  3 /**
  4  * @fileOverview Main application. Provides three kinds of services: a registrar, a template manager, and a notifier.
  5  */
  6 module.exports = App;
  7 
  8 var config = require('./config');
  9 var express = require('express');
 10 var http = require('http');
 11 var https = require('https');
 12 var APNs = require('./APNs');
 13 var GCM = require('./GCM');
 14 var Notifier = require('./notifier');
 15 var PayloadGenerator = require('./payloadGenerator');
 16 var Registrar = require('./registrar');
 17 var RegistrationStore = require('./registrationStore');
 18 var TemplateManager = require('./templateManager');
 19 var TemplateStore = require('./templateStore');
 20 var WNS = require('./WNS');
 21 
 22 /**
 23  * App constructor.
 24  *
 25  * @class
 26  *
 27  * An App object contains routes for HTTP/S requests to appropriate services.
 28  */
 29 function App()
 30 {
 31     this.log = config.log('App');
 32     this.registrationStoreName = config.registrations_table_name;
 33     this.templateStoreName = config.templates_table_name;
 34 }
 35 
 36 /**
 37  * App prototype.
 38  *
 39  * Defines the methods available to an App instance.
 40  */
 41 App.prototype = {
 42 
 43     /**
 44      * Initialize an App instance with the necessary services.
 45      *
 46      * @param {Function} callback the function to invoke when the App is initialized.
 47      */
 48     initialize: function(callback) {
 49         var log = this.log.child('initialize');
 50         var self = this;
 51         var awaiting = 2;
 52         var errors = null;
 53         var app = express();
 54         var registrationStore;
 55         var templateStore;
 56 
 57         var continuation = function(key, err) {
 58             var clog = log.child('continuation');
 59             clog.BEGIN(key, err);
 60 
 61             --awaiting;
 62             if (err !== null) {
 63                 clog.error(key, 'error:', err);
 64                 if (errors === null) {
 65                     errors = {};
 66                 }
 67                 errors[key] = err;
 68             }
 69 
 70             if (awaiting === 0) {
 71                 clog.info('creating notifier');
 72                 var senders = {'wns': new WNS(), 'apns': new APNs(), 'gcm': new GCM()};
 73                 var generator = new PayloadGenerator();
 74                 var service = new Notifier(templateStore, registrationStore, generator, senders);
 75                 self.notifier = service;
 76                 clog.info('adding bindings to notifier');
 77                 app.post('/post/:userId', service.postNotification.bind(service));
 78                 app.post('/logReceipt', service.logReceipt.bind(service));
 79                 clog.END();
 80 
 81                 callback(app, errors);
 82 
 83                 return;
 84             }
 85 
 86             clog.END('awaiting:', awaiting);
 87         };
 88 
 89         log.BEGIN();
 90 
 91         app.configure(
 92             function () {
 93                 app.use(app.router);
 94             });
 95 
 96         app.configure('development', function(){
 97                           app.use(express.errorHandler({dumpExceptions: true, showStack: true}));
 98                       });
 99 
100         app.configure('production', function(){
101                           app.use(express.errorHandler());
102                       });
103 
104         app.post('*', express.json());
105 
106         registrationStore = new RegistrationStore(this.registrationStoreName, function(err) {
107                 log.info('creating registrar');
108                 var service = self.registrar = new Registrar(registrationStore);
109                 log.info('adding bindings to registrar');
110                 app.get('/registrations/:userId', service.getRegistrations.bind(service));
111                 app.post('/registrations/:userId', service.addRegistration.bind(service));
112                 app.del('/registrations/:userId', service.deleteRegistration.bind(service));
113                 continuation('registrationStore', err);
114             });
115 
116         templateStore = new TemplateStore(this.templateStoreName, function(err) {
117                 log.info('creating template manager');
118                 var service = new TemplateManager(templateStore);
119                 log.info('adding bindings to template manager');
120                 app.get('/templates', service.getTemplates.bind(service));
121                 app.post('/templates', service.addTemplate.bind(service));
122                 app.del('/templates', express.json(), service.deleteTemplate.bind(service));
123                 continuation('templateStore', err);
124             });
125 
126         log.END();
127     }
128 };
129 
130 // Run if we are the main script
131 if (process.argv[1].indexOf('server.js') !== -1) {
132     var log = config.log('main');
133     var app = new App();
134     var port = process.env.PORT || 4465;
135 
136     process.on('uncaughtException', function (err) { console.log(err); });
137 
138     app.initialize(
139         function (app, errors) {
140             if (errors !== null) {
141                 log.error('failed to start due to errors during initialization: ', errors);
142             }
143             else {
144                 var server;
145                 if (config.ssl_authentication_enabled) {
146                     var options = {
147                         key: config.ssl_server_key,
148                         passphrase: config.ssl_server_key_passphrase,
149                         cert: config.ssl_server_certificate,
150                         ca: config.ssl_ca_certificate,
151                         requestCert: true,
152                         rejectUnauthorized: config.ssl_reject_unauthorized
153                         };
154                     server = https.createServer(options, app);
155                 }
156                 else {
157                     server = http.createServer(app);
158                 }
159                 log.info('starting server on port', port);
160                 server.listen(port);
161             }
162         }
163     );
164 }
