1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the APNs prototype and its methods.
  5  */
  6 module.exports = APNs;
  7 
  8 var config = require('./config');
  9 var apn = require('apn');
 10 var UserDeviceTracker = require('./userDeviceTracker');
 11 
 12 /**
 13  * APNs constructor.
 14  *
 15  * @class
 16  *
 17  * An APNs object sends JSON payloads to an APNs server for delivery to a specific iOS device.
 18  */
 19 function APNs(registrationStore) {
 20     this.log = config.log('APNs');
 21 
 22     this.registrationStore = registrationStore;
 23 
 24     /**
 25      * Cache of the N most recently seen userId / device token pairs.
 26      */
 27     this.userDeviceTracker = new UserDeviceTracker(config.apns_device_cache_size);
 28 
 29     // Terrible attempt to support a callback when notification gets sent out.
 30     this.lastSentNotification = null;
 31 
 32     /**
 33      * Error callback invoked whenever an error response comes over the connection. If the notification object is
 34      * valid and has a callback associated with it, invoke the callback.
 35      *
 36      * @param {Number} errNum numeric response returned by APNs server before it closed the connection
 37      * @param {apn.Notification} notification the Notification instance that triggered he error
 38      */
 39     var errorCallback = function (errNum, notification) {
 40         this.log.debug("APNs.errorCallback:", errNum, notification);
 41         this.lastSentNotification = null;
 42         if (notification !== null && typeof notification.callback === "function") {
 43             notification.callback({"retry":false, "invalidToken":errNum === 8});
 44         }
 45     }.bind(this);
 46 
 47     /**
 48      * Connection to APNs server.
 49      * @type apn.Connection
 50      */
 51     this.connection = new apn.Connection(
 52         {
 53             "cert": config.apns_client_certificate_file,
 54             "key": config.apns_client_private_key_file,
 55             "gateway": config.apns_service_host,
 56             "port": config.apns_service_port,
 57             "enhanced": true,
 58             "errorCallback": errorCallback,
 59             "cacheLength": 100
 60         }
 61     );
 62 
 63     /**
 64     * Override the socketDrained method for the connection. If there was a notification that was just sent out, then
 65     * invoke its callback if possible. NOTE: this is incredibly bogus because an error could fire later on for the
 66     * same notification.
 67     * @type Function
 68     */
 69     this.connection.socketDrained = function () {
 70         this.log.debug("APNs.socketDrained");
 71         if (this.lastSentNotification !== null) {
 72             var notification = this.lastSentNotification;
 73             if (notification !== null) {
 74                 this.lastSentNotification = null;
 75                 if (typeof notification.callback === "function") {
 76                     notification.callback({"retry":false, "invalidToken":false});
 77                 }
 78             }
 79         }
 80         apn.Connection.prototype.socketDrained.call(this.connection);
 81     }.bind(this);
 82 
 83     /**
 84      * Override the sendNotification method for the connection. Remember the notification being sent.
 85      * @type Function
 86      */
 87     this.connection.sendNotification = function (notification) {
 88         this.lastSentNotification = notification;
 89         apn.Connection.prototype.sendNotification.call(this.connection, notification);
 90     }.bind(this);
 91 
 92     if (typeof config.apns_feedback_interval !== 'undefined' && config.apns_feedback_interval > 0) {
 93         this.log.info('starting feedback service');
 94         this.feedback = new apn.Feedback(
 95             {
 96                 "cert": config.apns_client_certificate_file,
 97                 "key": config.apns_client_private_key_file,
 98                 "address": config.apns_feedback_host,
 99                 "port": config.apns_feedback_port,
100                 "enhanced": true,
101                 "errorCallback": errorCallback,
102                 "feedback": this.feedbackCallback.bind(this),
103                 "interval": 3600
104             });
105     }
106 }
107 
108 /**
109  * APNs prototype.
110  *
111  * Defines the methods available to an APNs instance.
112  */
113 APNs.prototype = {
114 
115     feedbackCallback: function (timestamp, device) {
116         var log = this.log.child('feedbackCallback');
117         var self = this;
118         log.BEGIN(timestamp, device);
119         var userId = this.userDeviceTracker.find(device.toString());
120         if (this.registrationStore && userId) {
121             log.info('removing registration for user ', userId, ' and device token ', device);
122             this.registrationStore.deleteRegistration(userId, device, function (err) {
123                 if (err) {
124                     log.error('failed to delete registration:', err);
125                 }
126             });
127         }
128         log.END();
129     },
130 
131     /**
132      * Send a notification payload to an APNs server.
133      *
134      * @param {NotificationRequest} req attributes of the notification to send
135      */
136     sendNotification: function(req) {
137         var log = this.log.child('sendNotification');
138         log.BEGIN(req);
139         var notification = new apn.Notification();
140         notification.callback = function (state) {
141             if (state.invalidToken) {
142                 req.removeRegistrationProc();
143             }
144         };
145 
146         notification.device = new apn.Device(new Buffer(req.token, 'base64'));
147         this.userDeviceTracker.add(req.userId, notification.device.toString());
148 
149         try {
150             notification.payload = JSON.parse(req.content.content);
151             log.debug('payload:', req.content.content);
152             this.connection.sendNotification(notification);
153         }
154         catch (err) {
155             log.error('failed to parse APN payload:', req.content.content);
156         }
157         log.END();
158     }
159 };
