1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the APNs prototype and its methods.
  5  */
  6 module.exports = APNs;
  7 
  8 var config = require("./config");
  9 var apn = require("apn");
 10 
 11 /**
 12  * APNs constructor.
 13  *
 14  * @class
 15  *
 16  * An APNs object sends JSON payloads to an APNs server for delivery to a specific iOS device.
 17  */
 18 function APNs() {
 19     this.log = config.log('APNs');
 20 
 21     // Terrible attempt to support a callback when notification gets sent out.
 22     this.lastSentNotification = null;
 23 
 24     /**
 25      * Error callback invoked whenever an error response comes over the connection. If the notification object is
 26      * valid and has a callback associated with it, invoke the callback.
 27      *
 28      * @param {Number} errNum numeric response returned by APNs server before it closed the connection
 29      * @param {apn.Notification} notification the Notification instance that triggered he error
 30      */
 31     var errorCallback = function (errNum, notification) {
 32         this.log.debug("APNs.errorCallback:", errNum, notification);
 33         this.lastSentNotification = null;
 34         if (notification !== null && typeof notification.callback === "function") {
 35             notification.callback({"retry":false, "invalidToken":errNum === 8});
 36         }
 37     }.bind(this);
 38 
 39     /**
 40      * Connection to APNs server.
 41      * @type apn.Connection
 42      */
 43     this.connection = new apn.Connection(
 44         {
 45             "cert": config.apns_client_certificate_file,
 46             "key": config.apns_client_private_key_file,
 47             "gateway": config.apns_service_host,
 48             "port": config.apns_service_port,
 49             "enhanced": true,
 50             "errorCallback": errorCallback,
 51             "cacheLength": 100
 52         }
 53     );
 54 
 55     /**
 56     * Override the socketDrained method for the connection. If there was a notification that was just sent out, then
 57     * invoke its callback if possible. NOTE: this is incredibly bogus because an error could fire later on for the
 58     * same notification.
 59     * @type Function
 60     */
 61     this.connection.socketDrained = function () {
 62         this.log.debug("APNs.socketDrained");
 63         if (this.lastSentNotification !== null) {
 64             var notification = this.lastSentNotification;
 65             if (notification !== null) {
 66                 this.lastSentNotification = null;
 67                 if (typeof notification.callback === "function") {
 68                     notification.callback({"retry":false, "invalidToken":false});
 69                 }
 70             }
 71         }
 72         apn.Connection.prototype.socketDrained.call(this.connection);
 73     }.bind(this);
 74 
 75     /**
 76      * Override the sendNotification method for the connection. Remember the notification being sent.
 77      * @type Function
 78      */
 79     this.connection.sendNotification = function (notification) {
 80         this.lastSentNotification = notification;
 81         apn.Connection.prototype.sendNotification.call(this.connection, notification);
 82     }.bind(this);
 83 }
 84 
 85 /**
 86  * APNs prototype.
 87  *
 88  * Defines the methods available to an APNs instance.
 89  */
 90 APNs.prototype = {
 91 
 92     /**
 93      * Send a notification payload to an APNs server.
 94      *
 95      * @param {NotificationRequest} req attributes of the notification to send
 96      */
 97     sendNotification: function(req) {
 98         var log = this.log.child('sendNotification');
 99         log.BEGIN(req);
100         var notification = new apn.Notification();
101         notification.callback = function (state) {
102             if (state.invalidToken) {
103                 req.removeRegistrationProc();
104             }
105         };
106         notification.device = new apn.Device(new Buffer(req.token, 'base64'));
107         notification.payload = JSON.parse(req.content.content);
108         log.debug('payload:', req.content.content);
109         this.connection.sendNotification(notification);
110         log.END();
111     }
112 };
