1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the UserDeviceTracker prototype and its methods.
  5  */
  6 module.exports = UserDeviceTracker;
  7 
  8 var Deque = require('./deque');
  9 var config = require('./config');
 10 
 11 /**
 12  * UserDeviceTracker constructor
 13  *
 14  * @class
 15  *
 16  * A UserDeviceTracker instance maintains a LIFO size-limited queue that contains Node instances that define a
 17  * relationship between a user ID and an APNs device token.
 18  */
 19 function UserDeviceTracker(maxSize) {
 20     this.log = config.log('UserDeviceTracker');
 21     this.maxSize = maxSize;
 22     this.active = new Deque();
 23     this.mapping = {};
 24 }
 25 
 26 /**
 27  * UserDeviceTracker.Node constructor
 28  *
 29  * @class
 30  *
 31  * A UserDeviceTracker.Node instance contains a user ID and a device token.
 32  */
 33 UserDeviceTracker.Node = function(userId, device) {
 34     this.userId = userId;
 35     this.device = device;
 36 };
 37 
 38 UserDeviceTracker.prototype = {
 39 
 40     /**
 41      * Create a new Node instance and add to the LIFO size-limited queue.
 42      *
 43      * @param {String} userId the user identifier to remember
 44      *
 45      * @param {String} device the iOS device token to remember
 46      */
 47     add: function (userId, device) {
 48         var log = this.log.child('add');
 49         log.BEGIN(userId, device);
 50 
 51         var node = this.mapping[device];
 52         if (typeof node !== 'undefined') {
 53             node.extricate();   // Unlink from the active queue
 54         }
 55         else {
 56             if (this.active.size() === this.maxSize) {
 57                 node = this.active.popBack();
 58                 delete this.mapping[node.data.device];
 59                 node.data.userId = userId;
 60                 node.data.device = device;
 61             }
 62             else {
 63                 node = new Deque.Node(new UserDeviceTracker.Node(userId, device));
 64             }
 65             this.mapping[device] = node;
 66         }
 67 
 68         this.active.pushFrontNode(node);
 69         log.END();
 70     },
 71 
 72     /**
 73      * Attempt to locate the user ID that belongs to a given iOS device token.
 74      *
 75      * @param {String} device the iOS device token to look for
 76      */
 77     find: function(device) {
 78         var log = this.log.child('find');
 79         log.BEGIN(device);
 80         var node = this.mapping[device];
 81         var value = node ? node.data.userId: null;
 82         if (value) {
 83             node.extricate();
 84             delete this.mapping[device];
 85         }
 86         log.END(value);
 87         return value;
 88     }
 89 };
