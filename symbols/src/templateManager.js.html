1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the TemplateManager prototype and its methods.
  5  */
  6 module.exports = TemplateManager;
  7 
  8 var HTTPStatus = require('http-status');
  9 var Model = require('model.js');
 10 
 11 /**
 12  * Initializes a new TemplateManager object. A TemplateManager stores and retrieves notification template definitions.
 13  * It relies on a TemplateStore object to provide the actual storage.
 14  *
 15  * @class TemplateManager
 16  *
 17  * @param {TemplateStore} templateStore the TemplateStore instance to rely on
 18  */
 19 function TemplateManager(templateStore) {
 20     this.log = require('./config').log('TemplateManager');
 21     this.templateStore = templateStore;
 22 
 23     this.FindKeyModel = Model.extend(
 24         {
 25             eventId: {required:true, type:'integer'},
 26             route: {required:true, type:'string'},
 27             templateVersion: {required:true, type:'string', minlength:1, maxlength:10},
 28             templateLanguage: {required:true, type:'string', minlength:2, maxlength:10},
 29             service: {required:true, type:'string', minlength:1, maxlength:10}
 30         }
 31     );
 32 
 33     this.TemplateKeyModel = this.FindKeyModel.extend(
 34         {
 35             notificationId: {required:true, type:'integer'}
 36         }
 37     );
 38 
 39     this.TemplateContentModel = Model.extend(
 40         {
 41             content: {required:true, type:'string'}
 42         }
 43     );
 44 
 45     this.TemplateDefinitionModel = this.TemplateKeyModel.extend(
 46         {
 47             template: {required:true, type:'templateContent'}
 48         },
 49         {
 50             types:{
 51                 templateContent:function (value) {
 52                     if (this.TemplateContentModel.validate(value) !== null) return null;
 53                     return value;
 54                 }.bind(this)
 55             }
 56         }
 57     );
 58 }
 59 
 60 TemplateManager.prototype = {
 61 
 62     /**
 63      * Add a template to the store.
 64      */
 65     addTemplate: function (req, res) {
 66         var log = this.log.child('addTemplate');
 67         log.BEGIN();
 68 
 69         var params = {
 70           eventId: req.param('eventId'),
 71           notificationId: req.param('notificationId'),
 72           route: req.param('route'),
 73           templateVersion: req.param('templateVersion'),
 74           templateLanguage: req.param('templateLanguage'),
 75           service: req.param('service'),
 76           template: req.param('template')
 77         };
 78 
 79         log.info('params:', params);
 80         var errors = this.TemplateDefinitionModel.validate(params);
 81         if (errors !== null) {
 82             log.error('invalid params:', JSON.stringify(errors));
 83             res.send(HTTPStatus.BAD_REQUEST);
 84             return;
 85         }
 86 
 87         if (typeof params.template.content !== 'string') {
 88             params.template.content = JSON.stringify(params.template.content);
 89         }
 90 
 91         var start = Date.now();
 92         this.templateStore.addTemplate(params, function (err, templateEntity)
 93         {
 94             var duration = Date.now() - start;
 95             if (err) {
 96                 log.error('TemplateStore.addTemplate error:', err);
 97                 res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
 98             }
 99             else {
100                 var tmp = {
101                     'tableStoreDuration': duration
102                 };
103                 res.json(HTTPStatus.OK, tmp);
104             }
105 
106             log.END(duration, 'msec');
107         });
108     },
109 
110     /**
111      * Get a template.
112      */
113     getTemplates: function (req, res) {
114         var log = this.log.child('getTemplates');
115         log.BEGIN(req.query);
116 
117         var params = {
118           eventId: req.param('eventId'),
119           route: req.param('route'),
120           templateVersion: req.param('templateVersion'),
121           templateLanguage: req.param('templateLanguage'),
122           service: req.param('service')
123         };
124 
125         log.info('params:', params);
126 
127         var errors = this.FindKeyModel.validate(params);
128         if (errors !== null) {
129             log.error('invalid params:', JSON.stringify(errors));
130             res.send(HTTPStatus.BAD_REQUEST);
131             return;
132         }
133 
134         var reg = {
135             'templateVersion': '' + params.templateVersion,
136             'templateLanguage': params.templateLanguage,
137             'service': params.service,
138             'routes': [ {'name':params.route, 'token':''} ]
139         };
140 
141         var start = Date.now();
142         this.templateStore.findTemplates(params.eventId, [reg], function (err, templates)
143         {
144             var duration = Date.now() - start;
145             if (err) {
146                 log.error('TemplateStore.findTemplates error:', err);
147                 res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
148             }
149             else if (templates.length === 0) {
150                 log.info('no templates found in store');
151                 res.send(HTTPStatus.NOT_FOUND);
152             }
153             else {
154                 var tmp = {
155                     'templates': templates,
156                     'tableStoreDuration': duration
157                 };
158                 res.json(HTTPStatus.OK, tmp);
159             }
160 
161             log.END(duration, 'msec');
162         });
163     },
164 
165     /**
166      * Delete a template.
167      */
168     deleteTemplate: function (req, res) {
169         var log = this.log.child('deleteTemplate');
170         log.BEGIN();
171 
172         var params = {
173           eventId: req.param('eventId'),
174           notificationId: req.param('notificationId'),
175           route: req.param('route'),
176           templateVersion: req.param('templateVersion'),
177           templateLanguage: req.param('templateLanguage'),
178           service: req.param('service')
179         };
180 
181         log.info('params:', params);
182 
183         var errors = this.TemplateKeyModel.validate(params);
184         if (errors !== null) {
185             log.error('invalid params:', JSON.stringify(errors));
186             res.send(HTTPStatus.BAD_REQUEST);
187             return;
188         }
189 
190         var start = Date.now();
191         this.templateStore.removeTemplate(params, function (err, templateEntity)
192         {
193             var duration = Date.now() - start;
194             if (err !== null) {
195                 log.error('TemplateStore.removeTemplate error:', err);
196                 res.send(HTTPStatus.NOT_FOUND);
197             }
198             else {
199                 log.info('removed');
200                 res.send(HTTPStatus.NO_CONTENT);
201             }
202             log.END(duration, 'msecs');
203         });
204     }
205 };
