1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the TemplateManager prototype and its methods.
  5  */
  6 module.exports = TemplateManager;
  7 
  8 var express = require('express');
  9 var HTTPStatus = require('http-status');
 10 var Model = require('model.js');
 11 
 12 /**
 13  * TemplateManager constructor.
 14  *
 15  * @class
 16  *
 17  * A TemplateManager stores and retrieves notification template definitions. It relies on a TemplateStore object to
 18  * provide the actual storage.
 19  *
 20  * @param {TemplateStore} templateStore the TemplateStore instance to rely on for persistent data operations.
 21  */
 22 function TemplateManager(templateStore) {
 23     this.log = require('./config').log('TemplateManager');
 24     this.templateStore = templateStore;
 25 
 26     /**
 27      * JSON schema for query to locate a specific template. Attributes:
 28      *
 29      * - eventId: the notification event identifier associated with the template
 30      * - route: the of the route associated with the template.
 31      * - templateVersion: the version of the template to locate
 32      * - templateLanguage: the language of the template to locate
 33      * - service: the OS-specific service that the template applies to
 34      */
 35     this.FindKeyModel = Model.extend(
 36         {
 37             eventId: {required:true, type:'integer'},
 38             route: {required:true, type:'string'},
 39             templateVersion: {required:true, type:'string', minlength:1, maxlength:10},
 40             templateLanguage: {required:true, type:'string', minlength:2, maxlength:10},
 41             service: {required:true, type:'string', minlength:1, maxlength:10}
 42         }
 43     );
 44 
 45     /**
 46      * JSON schema for a unique template key. Derives from {@link TemplateManager#FindKeyModel}. Attributes:
 47      *
 48      * - notificationId: the unique identifier associated with this template
 49      */
 50     this.TemplateKeyModel = this.FindKeyModel.extend(
 51         {
 52             notificationId: {required:true, type:'integer'}
 53         }
 54     );
 55 
 56     /**
 57      * JSON schema for template contents. Attributes:
 58      *
 59      * - content: the template content of the notification that will be sent to devices.
 60      */
 61     this.TemplateContentModel = Model.extend(
 62         {
 63             content: {required:true, type:'string'}
 64         }
 65     );
 66 
 67     /**
 68      * JSON schema for template definitions. Derived from the {@link TemplateManager#TemplateKeyModel}. Attributes:
 69      *
 70      * - template: the template content of the notification that will be sent to devices.
 71      */
 72     this.TemplateDefinitionModel = this.TemplateKeyModel.extend(
 73         {
 74             template: {required:true, type:'templateContent'}
 75         },
 76         {
 77             types:{
 78                 templateContent:function (value) {
 79                     if (this.TemplateContentModel.validate(value) !== null) return null;
 80                     return value;
 81                 }.bind(this)
 82             }
 83         }
 84     );
 85 }
 86 
 87 /**
 88  * TemplateManager prototype.
 89  *
 90  * Defines the methods available to a TemplateManager instance.
 91  */
 92 TemplateManager.prototype = {
 93 
 94     /**
 95      * Add Express routes for the Registrar API.
 96      *
 97      * @param {Express.App} app the application to route
 98      */
 99     route: function(app) {
100         var log = this.log.child('route');
101         log.BEGIN('adding bindings');
102         app.get('/templates', this.get.bind(this));
103         app.post('/templates', this.set.bind(this));
104         app.del('/templates', this.del.bind(this));
105         log.END();
106     },
107 
108     /**
109      * Get one or more existing templates that match a given key. The key is made up of the following parameters:
110      *
111      * - eventId: the notification event identifier associated with the template
112      * - route: the of the route associated with the template.
113      * - templateVersion: the version of the template to locate
114      * - templateLanguage: the language of the template to locate
115      * - service: the OS-specific service that the template applies to
116      *
117      * HTTP Responses:
118      *
119      * - 400 BAD REQUEST: missing or invalid query attribute(s)
120      * - 500 INTERNAL SERVER ERROR: unable to process the request for some reason
121      * - 404 NOT FOUND: no templates were found for the given key attributes
122      * - 200 OK: one or more templates were found
123      *
124      * @param req Contents of the incoming request.
125      *
126      * @param res Response generator for the request. Any templates found will be returned in JSON output.
127      *
128      */
129     get: function (req, res) {
130         var log = this.log.child('get');
131         log.BEGIN(req.query);
132 
133         var params = {
134           eventId: req.param('eventId'),
135           route: req.param('route'),
136           templateVersion: req.param('templateVersion'),
137           templateLanguage: req.param('templateLanguage'),
138           service: req.param('service')
139         };
140 
141         log.info('params:', params);
142 
143         var errors = this.FindKeyModel.validate(params);
144         if (errors !== null) {
145             log.error('invalid params:', JSON.stringify(errors));
146             res.send(HTTPStatus.BAD_REQUEST);
147             return;
148         }
149 
150         var reg = {
151             'templateVersion': '' + params.templateVersion,
152             'templateLanguage': params.templateLanguage,
153             'service': params.service,
154             'routes': [ {'name':params.route, 'token':''} ]
155         };
156 
157         var start = Date.now();
158         this.templateStore.findTemplates(params.eventId, [reg], function (err, templates) {
159             var duration = Date.now() - start;
160             if (err) {
161                 log.error('TemplateStore.findTemplates error:', err);
162                 res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
163             }
164             else if (templates.length === 0) {
165                 log.info('no templates found in store');
166                 res.send(HTTPStatus.NOT_FOUND);
167             }
168             else {
169                 var tmp = {
170                     'templates': templates,
171                     'tableStoreDuration': duration
172                 };
173                 res.json(HTTPStatus.OK, tmp);
174             }
175 
176             log.END(duration, 'msec');
177         });
178     },
179 
180     /**
181      * Add or update a template with a given key. Expects the following parameters:
182      *
183      * - eventId: the notification event identifier associated with the template
184      * - notificationId: the unique identifier associated with the template
185      * - route: the of the route associated with the template.
186      * - templateVersion: the version of the template to locate
187      * - templateLanguage: the language of the template to locate
188      * - service: the OS-specific service that the template applies to
189      * - template: JSON object that defines the template payload
190      *
191      * HTTP Responses:
192      *
193      * - 400 BAD REQUEST: missing or invalid template attribute(s)
194      * - 500 INTERNAL SERVER ERROR: unable to process the request for some reason
195      * - 200 OK: one or moret templates were found
196      *
197      * @param req Contents of the incoming request.
198      *
199      * @param res Response generator for the request.
200      */
201     set: function (req, res) {
202         var log = this.log.child('set');
203         log.BEGIN();
204 
205         var params = {
206             eventId: req.param('eventId'),
207             notificationId: req.param('notificationId'),
208             route: req.param('route'),
209             templateVersion: req.param('templateVersion'),
210             templateLanguage: req.param('templateLanguage'),
211             service: req.param('service'),
212             template: req.param('template')
213         };
214 
215         log.info('params:', params);
216         var errors = this.TemplateDefinitionModel.validate(params);
217         if (errors !== null) {
218             log.error('invalid params:', JSON.stringify(errors));
219             res.send(HTTPStatus.BAD_REQUEST);
220             return;
221         }
222 
223         if (typeof params.template.content !== 'string') {
224             params.template.content = JSON.stringify(params.template.content);
225         }
226 
227         var start = Date.now();
228         this.templateStore.addTemplate(params, function (err, templateEntity)
229         {
230             var duration = Date.now() - start;
231             if (err) {
232                 log.error('TemplateStore.addTemplate error:', err);
233                 res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
234             }
235             else {
236                 var tmp = {
237                     'tableStoreDuration': duration
238                 };
239                 res.json(HTTPStatus.OK, tmp);
240             }
241 
242             log.END(duration, 'msec');
243         });
244     },
245 
246     /**
247      * Delete a template from the template store. Expects the following parameters:
248      *
249      * - eventId: the notification event identifier associated with the template
250      * - notificationId: the unique identifier associated with the template
251      * - route: the of the route associated with the template.
252      * - templateVersion: the version of the template to locate
253      * - templateLanguage: the language of the template to locate
254      * - service: the OS-specific service that the template applies to
255      *
256      * HTTP Responses:
257      *
258      * - 400 BAD REQUEST: missing or invalid template attribute(s)
259      * - 404 NOT FOUND: unable to find the template to delete
260      * - 204 NO CONTENT: found and deleted the template
261      *
262      * @param req Contents of the incoming request.
263      *
264      * @param res Response generator for the request.
265      *
266      */
267     del: function (req, res) {
268         var log = this.log.child('del');
269         log.BEGIN();
270 
271         var params = {
272             eventId: req.param('eventId'),
273             notificationId: req.param('notificationId'),
274             route: req.param('route'),
275             templateVersion: req.param('templateVersion'),
276             templateLanguage: req.param('templateLanguage'),
277             service: req.param('service')
278         };
279 
280         log.info('params:', params);
281 
282         var errors = this.TemplateKeyModel.validate(params);
283         if (errors !== null) {
284             log.error('invalid params:', JSON.stringify(errors));
285             res.send(HTTPStatus.BAD_REQUEST);
286             return;
287         }
288 
289         var start = Date.now();
290         this.templateStore.removeTemplate(params, function (err, templateEntity) {
291             var duration = Date.now() - start;
292             if (err !== null) {
293                 log.error('TemplateStore.removeTemplate error:', err);
294                 res.send(HTTPStatus.NOT_FOUND);
295             }
296             else {
297                 log.info('removed');
298                 res.send(HTTPStatus.NO_CONTENT);
299             }
300             log.END(duration, 'msecs');
301         });
302     }
303 };
