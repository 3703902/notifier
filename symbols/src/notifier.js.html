1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Notifier prototype and its methods.
  5  */
  6 module.exports = Notifier;
  7 
  8 var HTTPStatus = require('http-status');
  9 var Model = require('model.js');
 10 var PostTracker = require('./postTracker');
 11 var NotificationRequest = require('./notificationRequest');
 12 
 13 /**
 14  * Notifier constructor
 15  *
 16  * @class
 17  *
 18  * A Notifier pairs notification templates to user registrations and emits those that match.
 19  *
 20  * @param {TemplateStore} templateStore the repository of notification templates to rely on
 21  * @param {RegistrationStore} registrationStore the repository of user registrations to rely on
 22  * @param {PayloadGenerator} generator the notification generator to use for template processing
 23  * @param {Hash} senders a mapping of notification service tags to objects that provide a sendNotification method
 24  */
 25 function Notifier(templateStore, registrationStore, generator, senders) {
 26     this.log = require('./config').log('notifier');
 27     this.templateStore = templateStore;
 28     this.registrationStore = registrationStore;
 29     this.generator = generator;
 30     this.senders = senders;
 31 
 32     this.sequenceId = 1;
 33     this.postTracker = new PostTracker(100);
 34 
 35     this.PostModel = Model.extend(
 36         {
 37             userId: {required:true, type:'string', minlength:2, maxlength:128},
 38             eventId: {required:true, type:'integer'},
 39             substitutions: {type:'stringmap'},
 40             tokens: {type: 'array'}
 41         },
 42         {
 43             types:
 44             {
 45                 stringmap: function (value) {
 46                     if (typeof value !== 'object') return null;
 47                     for (var key in value) {
 48                         if (typeof value[key] !== 'string') return null;
 49                     }
 50                     return value;
 51                 }
 52             }
 53         }
 54     );
 55 
 56     this.LogReceiptModel = Model.extend(
 57         {
 58             sequenceId: {required:true, type:'integer'},
 59             when: {required:true, type:'date'}
 60         }
 61     );
 62 }
 63 
 64 /**
 65  * Notifier prototype.
 66  *
 67  * Defines the methods available to an APNs instance.
 68  */
 69 Notifier.prototype = {
 70 
 71     /**
 72      * Post a notification to a user.
 73      *
 74      * @param {Request} req incoming HTTP request settings and values
 75      *
 76      * @param {Response} res outgoing HTTP response values
 77      */
 78     postNotification: function(req, res) {
 79         var self = this, duration;
 80         var log = self.log.child('postNotification');
 81         var result = {count: 0};
 82         log.BEGIN();
 83 
 84         var params = {
 85             userId: req.param('userId'),
 86             eventId: req.param('eventId'),
 87             substitutions: req.param('substitutions'),
 88             tokens: req.param('tokens')
 89         };
 90 
 91         log.info('params:', params);
 92 
 93         var errors = this.PostModel.validate(params);
 94         if (errors !== null) {
 95             log.error('invalid params:', JSON.stringify(errors));
 96             res.send(HTTPStatus.BAD_REQUEST);
 97             return;
 98         }
 99 
100         var substitutions = params.substitutions;
101         if (typeof substitutions !== 'object') {
102             substitutions = {};
103             params.substitutions = substitutions;
104         }
105 
106         var start = Date.now();
107         log.debug('start:', start);
108 
109         var sequenceId = ++this.sequenceId;
110         substitutions.SEQUENCE_ID = sequenceId;
111         this.postTracker.add(sequenceId, start);
112 
113         // Fetch the registrations for the given user
114         this.registrationStore.getRegistrations(params.userId, params.tokens, function(err, registrations) {
115             if (err !== null) {
116                 log.error('RegistrationStore.getRegistrations error:', err);
117                 res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
118                 return;
119             }
120 
121             if (registrations.length === 0) {
122                 log.info('no registrations exist for user', params.userId);
123                 res.send(HTTPStatus.NO_CONTENT);
124                 duration = Date.now() - start;
125                 log.END(duration, 'msecs');
126                 return;
127             }
128 
129             // Find the templates that apply for the given registrations
130             self.templateStore.findTemplates(params.eventId, registrations, function(err, matches) {
131                 var token = null;
132                 if (err !== null) {
133                     log.error('TemplateStore.findTemplates error:', err);
134                     res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
135                     duration = Date.now() - start;
136                     log.END(duration, 'msecs');
137                     return;
138                 }
139 
140                 if (matches.length === 0) {
141                     log.info('TemplateStore.findTemplates returned no matches');
142                 }
143 
144                 result.count = matches.length;
145                 res.json(HTTPStatus.ACCEPTED, result);
146 
147                 var removeRegistrationProc = function () {
148                     log.debug('callback:', params.userId, token, result);
149                     self.registrationStore.deleteRegistrationEntity(params.userId, token,
150                                                                     function (err) {
151                                                                         log.error('removeCallback:', err);
152                                                                     });
153                 };
154 
155                 // For each template, generate a notification payload and send it on its way.
156                 for (var index in matches) {
157                     var match = matches[index];
158                     var template = match.template;
159                     template.content = self.generator.transform(template.content, params.substitutions);
160                     token = match.token;
161                     self.deliver(match.service, new NotificationRequest(params.userId, token, template, 30,
162                                                                         removeRegistrationProc));
163                 }
164 
165                 duration = Date.now() - start;
166                 log.END(duration, 'msecs');
167             });
168         });
169     },
170 
171     logReceipt: function(req, res) {
172         var log = this.log.child('logReceipt');
173         log.BEGIN();
174 
175         var params = {
176             sequenceId: req.param('sequenceId'),
177             when: req.param('when')
178         };
179 
180         log.info('params:', params);
181 
182         var errors = this.LogReceiptModel.validate(params);
183         if (errors !== null) {
184             log.error('invalid params:', errors);
185             res.send(HTTPStatus.BAD_REQUEST);
186             return;
187         }
188 
189         this.postTracker.track(Number(params.sequenceId), new Date(params.when));
190         res.send(HTTPStatus.NO_CONTENT); // OK but no content
191 
192         log.END();
193     },
194 
195     /**
196      * Deliver a generated notification to a specific service.
197      *
198      * @param {String} service the notification service tag
199      *
200      * @param {NotificationRequest} request description of the request to send
201      */
202     deliver: function(service, request) {
203         this.senders[service].sendNotification(request);
204     }
205 };
