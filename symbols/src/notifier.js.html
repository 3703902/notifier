1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Notifier prototype and its methods.
  5  */
  6 module.exports = Notifier;
  7 
  8 var express = require('express');
  9 var HTTPStatus = require('http-status');
 10 var Model = require('model.js');
 11 var PostTracker = require('./postTracker');
 12 
 13 /**
 14  * Notifier constructor
 15  *
 16  * @class
 17  *
 18  * A Notifier pairs notification templates to user registrations and emits those that match. One notification event can
 19  * spawn multiple actual notification deliveries depending on what registrations are present for the user and what
 20  * templates map to a notification event ID.
 21  *
 22  * @param {TemplateStore} templateStore the repository of notification templates to rely on
 23  * @param {RegistrationStore} registrationStore the repository of user registrations to rely on
 24  * @param {Hash} senders a mapping of notification service tags to objects that provide a sendNotification method
 25  */
 26 function Notifier(templateStore, registrationStore, senders, monitorManager) {
 27     this.log = require('./config').log('Notifier');
 28     this.log.BEGIN()
 29 
 30     /**
 31      * The backing store for templates. Notifier asks it for templates associated with a given event ID.
 32      *
 33      * @type TemplateStore
 34      */
 35     this.templateStore = templateStore;
 36 
 37     /**
 38      * The backing store for registrations. Notifier asks it for registrations associated with a given user ID. Also,
 39      * if a notification sender object encounters a registration that is no longer active, it will be used to remove
 40      * the registration.
 41      *
 42      * @type RegistrationStore
 43      */
 44     this.registrationStore = registrationStore;
 45 
 46     /**
 47      * Mapping of service tags ("wns", "apns", "gcm") to objects that can deliver a notification to a device.
 48      *
 49      * @type Hash
 50      */
 51     this.senders = senders;
 52 
 53     this.monitorManager = monitorManager;
 54 
 55     /**
 56      * Sequence ID for tracking. This value is added to all substitution maps under the name "SEQUENCE_ID"
 57      *
 58      * @type Integer
 59      */
 60     this.sequenceId = 1;
 61 
 62     /**
 63      * PostTracker instance that remembers notification requests by sequenceId value. Used by the logReceipt method to
 64      * calculate notification round-trip times.
 65      */
 66     this.postTracker = new PostTracker(100);
 67 
 68     /**
 69      * JSON schema for filter values. A filter allows one to limit which registrations will be used for notification
 70      * delivery. The attribute names closely match those in {@link Registrar#RegistrationModel}, though route
 71      * attributes appear with a 'route' prefix. Each attribute has as its value an array which contains one or values
 72      * that determine which registration values are acceptable for use in a delivery.
 73      *
 74      * - registrationId: array of acceptable registrationId values
 75      * - templateVersion: array of acceptable template version values
 76      * - templateLanguage: array of acceptable template language values
 77      * - service: array of acceptable notification deliver service values
 78      * - routeName: array of acceptable route name values
 79      * - routeToken: array of acceptable route token values
 80      *
 81      * All attributes are optional.
 82      *
 83      * @type Model
 84      */
 85     this.FilterModel = Model.extend(
 86         {
 87             registrationId: {type: 'array'},
 88             templateVersion: {type: 'array'},
 89             templateLanguage: {type: 'array'},
 90             service: {type: 'array'},
 91             routeName: {type: 'array'},
 92             routeToken: {type: 'array'}
 93         }
 94     );
 95 
 96     /**
 97      * JSON schema for a notification posting request. A notification goes to a specific user. The request may contain
 98      * a mapping of template placeholder names to runtime values. This mapping is used during notification payload
 99      * generation, where a template placeholder gets substituted with a value from the substitutions mapping. Finally,
100      * there may be a filter definition (see {@link Notifier#FilterModel}) that will restrict notification delivery to
101      * a subset of the registrations for the user.
102      *
103      * - userId: the user to notify
104      * - eventId: the event ID to notify about
105      * - substitutions: the optional mapping used to perform template placeholder substitution
106      * - filter: an optional filter definition to restrict registration use
107      *
108      * @type Model
109      */
110     this.PostModel = Model.extend(
111         {
112             userId: {required:true, type:'string', minlength:2, maxlength:128},
113             eventId: {required:true, type:'integer'},
114             substitutions: {type:'stringmap'},
115             filter: {type: 'filter'}
116         },
117         {
118             types:
119             {
120                 stringmap: function (value) {
121                     if (typeof value !== 'object') return null;
122                     for (var key in value) {
123                         if (typeof value[key] !== 'string') return null;
124                     }
125                     return value;
126                 },
127                 filter: function (value) {
128                     var log = this.log.child('filter model');
129                     var err = this.FilterModel.validate(value);
130                     log.debug('filter validation:', err);
131                     if (err !== null) return null;
132                     return value;
133                 }.bind(this)
134             }
135         }
136     );
137 
138     /**
139      * JSON schema for a logReceipt call.
140      *
141      * - sequenceId: the sequence ID associated with the notification request
142      * - when: the timestamp when the notification arrived at a client
143      */
144     this.LogReceiptModel = Model.extend(
145         {
146             sequenceId: {required:true, type:'integer'},
147             when: {required:true, type:'date'}
148         }
149     );
150 }
151 
152 Notifier.prototype = {
153 
154     /**
155      * Add Express routes for the Notifier API.
156      *
157      * @param {Express.App} app the application to route HTTP requests.
158      */
159     route: function (app) {
160         var log = this.log.child('route');
161         log.BEGIN('adding bindings');
162         app.post('/post/:userId', express.json(), this.postNotification.bind(this));
163         log.END();
164     },
165 
166     /**
167      * Post a notification to a user.
168      *
169      * @param {Express.Request} req Describes the incoming HTTP request. Message body must be JSON that follows the
170      * {@link Notifier#PostModel} model.
171      *
172      * @param {Express.Response} res HTTP response generator for the request. Outputs JSON if no error.
173      *
174      * - 204 NO CONTENT: accepted request and queued notification delivery
175      * - 400 BAD REQUEST: missing or invalid attribute(s)
176      * - 500 INTERNAL SERVER ERROR: unable to obtain registrations and/or templates to work with
177      */
178     postNotification: function (req, res) {
179         var self = this, duration;
180         var log = self.log.child('postNotification');
181         var result = {count: 0};
182         log.BEGIN();
183 
184         var params = {
185             userId: req.param('userId'),
186             eventId: req.param('eventId'),
187             substitutions: req.param('substitutions'),
188             filter: req.param('filter')
189         };
190 
191         log.info('params:', params);
192 
193         var errors = this.PostModel.validate(params);
194         if (errors !== null) {
195             log.error('invalid params:', JSON.stringify(errors));
196             log.error(params);
197             res.send(HTTPStatus.BAD_REQUEST);
198             return;
199         }
200 
201         var substitutions = params.substitutions;
202         if (typeof substitutions !== 'object') {
203             substitutions = {};
204             params.substitutions = substitutions;
205         }
206 
207         var start = Date.now();
208         log.debug('start:', start);
209 
210         self.monitorManager.post(params.userId, 'posting notification to user');
211         self.monitorManager.post(params.userId, 'params: ' + JSON.stringify(params));
212 
213         var sequenceId = ++this.sequenceId;
214         substitutions.SEQUENCE_ID = sequenceId;
215         this.postTracker.add(sequenceId, start);
216 
217         // Fetch the registrations for the given user
218         this.registrationStore.get(params.userId, params.filter, function (err, registrations) {
219             if (err !== null) {
220                 log.error('RegistrationStore.get error:', err);
221                 self.monitorManager.post(params.userId, 'failed to get registrations for user');
222                 res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
223                 return;
224             }
225 
226             if (registrations.length === 0) {
227                 log.info('no registrations exist for user', params.userId);
228                 self.monitorManager.post(params.userId, 'no registrations found for user');
229                 res.json(HTTPStatus.ACCEPTED, result);
230                 duration = Date.now() - start;
231                 log.END(duration, 'msecs');
232                 return;
233             }
234 
235             // Find the templates that apply for the given registrations
236             self.templateStore.find(params.eventId, registrations, function (err, matches) {
237                 var token = null;
238                 if (err !== null) {
239                     log.error('TemplateStore.find error:', err);
240                     self.monitorManager.post(params.userId, 'failed to get templates');
241                     res.send(HTTPStatus.INTERNAL_SERVER_ERROR);
242                     duration = Date.now() - start;
243                     log.END(duration, 'msecs');
244                     return;
245                 }
246 
247                 if (matches.length === 0) {
248                     log.info('TemplateStore.find returned no matches');
249                     self.monitorManager.post(params.userId, 'no templates matching registrations');
250                 }
251 
252                 self.monitorManager.post(params.userId, 'sending ' + matches.length + ' notifications');
253 
254                 // For each template, generate a notification payload and send it on its way.
255                 for (var index in matches) {
256                     var notificationRequest = matches[index];
257                     log.info(notificationRequest);
258                     notificationRequest.prepare(substitutions, params.userId, self.registrationStore, 30);
259                     log.debug('*** service:', notificationRequest.service);
260                     self.senders[notificationRequest.service].sendNotification(notificationRequest);
261                 }
262 
263                 duration = Date.now() - start;
264                 log.END(duration, 'msecs');
265                 result.count = matches.length;
266                 res.json(HTTPStatus.ACCEPTED, result);
267                 return;
268             });
269         });
270     }
271 };
