  3 module.exports = MonitorManager;
  4 
  5 var azure = require('azure');
  6 var Q = require('q');
  7 var config = require('./config');
  8 
  9 function MonitorManager(topicName, isServer, callback) {
 10     var self = this;
 11 
 12     var log = this.log = config.log('MonitorManager');
 13     log.BEGIN();
 14 
 15     if (! topicName) {
 16         topicName = config.monitor_topic_name;
 17     }
 18 
 19     self.topicName = topicName;
 20     self.serviceBusService = azure.createServiceBusService();
 21     self.topicConfig = { MaxSizeInMegabytes: '1024', DefaultMessageTimeToLive: 'PT1M' };
 22 
 23     self.userMonitoringExpirations = {};
 24     self.userMonitoringExpirations[config.monitor_command_channel_name] = false;
 25 
 26     self.postPromiseChain = Q();
 27     self.registrationMonitor = null;
 28 
 29     self.createTopic(isServer, callback);
 30 }
 31 
 32 function Monitor(manager, name, userId) {
 33     var log = this.log = config.log('Monitor ' + userId);
 34     this.mgr = manager;
 35     this.userId = userId;
 36     this.name = name;
 37     this.socket = null;
 38     this.expiration = null;
 39 };
 40 
 41 Monitor.prototype = {
 42 
 43     register: function() {
 44         var self = this;
 45         var log = self.log.child('register');
 46         log.BEGIN();
 47         if (self.mgr) {
 48             self.mgr.registerMonitor(self);
 49         }
 50         log.END();
 51     },
 52 
 53     unregister: function() {
 54         var self = this;
 55         var log = self.log.child('unregister');
 56         log.BEGIN();
 57 
 58         if (self.mgr) {
 59             self.mgr.unregisterMonitor(self);
 60             self.mgr = null;
 61         }
 62 
 63         log.END();
 64     }
 65 };
 66 
 67 MonitorManager.prototype = {
 68 
 69     addSocket: function(userId, socket) {
 70         var self = this;
 71         var log = self.log.child('addSocket');
 72         self.makeMonitor(userId, function(err, monitor) {
 73             if (err) {
 74                 log.error('failed to create a monitor for the socket');
 75             }
 76             else {
 77                 log.debug('using monitor', monitor.name);
 78 
 79                 socket.on('disconnect', function () {
 80                     log.error('remote socket disconnected');
 81                     monitor.socket = null;
 82                     monitor.unregister();
 83                 });
 84 
 85                 monitor.socket = socket;
 86                 socket.monitor = monitor;
 87 
 88                 self.readLoop(monitor, function (err, line) {
 89                     if (! monitor.socket || ! monitor.mgr) return false;
 90                     log.debug('sending to web client');
 91                     monitor.socket.emit('line', line);
 92                     return true;
 93                 });
 94             }
 95             log.END();
 96         });
 97     },
 98 
 99     removeSocket: function(userId, socket) {
100         var self = this;
101         var log = self.log.child('removeSocket');
102         log.BEGIN('userId:', userId, 'monitor:', socket.monitor);
103         self.unregisterMonitor(socket.monitor);
104     },
105 
106     makeMonitor: function(userId, callback) {
107         var self = this;
108         var log = self.log.child('makeMonitor');
109         var name = 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
110             var r = Math.random()*16|0, v = c == 'x' ? r : (r&0x3|0x8);
111             return v.toString(16);
112         });
113 
114         log.BEGIN('name:', name);
115         log.BEGIN('userId:', userId);
116 
117         // Create the new subscription
118         //
119         self.serviceBusService.createSubscription(self.topicName, name,
120             function (err) {
121                 if (err) {
122                     log.error('failed to create subscription:', err);
123                     callback(err, null);
124                     return;
125                 }
126 
127                 log.debug('created subscription');
128 
129                 // Subscription created. Remove the default MatchAll rule
130                 self.serviceBusService.deleteRule(self.topicName,
131                                                   name,
132                                                   azure.Constants.ServiceBusConstants.DEFAULT_RULE_NAME,
133                     function (err) {
134                         if (err) log.error(err);
135                         log.debug('removed existing rule');
136                         var options = { sqlExpressionFilter: "userId = '" + userId + "'"};
137 
138                         // Create our own filter to match only the given user ID
139                         self.serviceBusService.createRule(self.topicName,
140                                                           name,
141                                                           'filter',
142                                                           options,
143                             function (err) {
144                                 if (err) {
145                                     log.error('failed to create filter:', err);
146                                     callback(err, null);
147                                 }
148                                 else {
149                                     var monitor = new Monitor(self, name, userId);
150                                     log.debug('created new rule filtering on', userId);
151                                     self.registerMonitor(monitor, function (err) {
152                                         callback(err, monitor);
153                                     });
154                                 }
155                             });
156                     });
157             });
158     },
159 
160     registerMonitor: function(monitor, callback) {
161         var self = this;
162         var log = self.log.child('registerMonitor');
163         log.BEGIN(monitor.userId, monitor.name);
164         if (monitor.userId == config.monitor_command_channel_name) {
165             log.END();
166             if (callback) callback(null);
167             return;
168         }
169 
170         self.post(config.monitor_command_channel_name, '+' + monitor.userId, function (err) {
171             if (err) log.error('failed to post monitor command for', monitor.userId, err);
172             log.END();
173             if (callback) callback(err);
174         });
175     },
176 
177     unregisterMonitor: function(monitor, callback) {
178         var self = this;
179         var log = self.log.child('unregisterMonitor');
180         log.BEGIN(monitor.userId, monitor.name);
181         if (monitor.userId == config.monitor_command_channel_name) {
182             log.END();
183             if (callback) callback(null);
184             return;
185         }
186 
187         self.post(monitor.userId, '*** monitoring stopped ***', function (err) {
188             self.post(config.monitor_command_channel_name, '-' + monitor.userId, function (err) {
189                 if (err) log.error('failed to post forget command for', monitor.userId, err);
190                 log.debug('deleting subscription for', monitor.name);
191                 self.serviceBusService.deleteSubscription(self.topicName, monitor.name, function (err) {
192                     if (err) log.error('failed to delete subscription:', err);
193                     log.END();
194                     if (callback) callback(err);
195                 });
196             });
197         });
198     },
199 
200     post: function(userId, line, callback) {
201         var self = this;
202         var log = self.log.child('post');
203         var expiration = self.userMonitoringExpirations[userId];
204         var proc, now;
205 
206         log.BEGIN('userId:', userId, 'line:', line, 'expiration:', expiration);
207 
208         // Silently drop anything that is not actively being monitored
209         //
210         if (expiration === undefined) {
211             if (callback) callback(null);
212             return;
213         }
214 
215         log.BEGIN(userId, line);
216 
217         // Check if the monitoring has expired
218         //
219         now = new Date();
220         if (expiration && expiration < now) {
221             line = '*** monitoring for has expired ***';
222             delete self.userMonitoringExpirations[userId];
223         }
224 
225         proc = function () {
226             var deferred = Q.defer();
227             var message = { 'body': line, 'customProperties': { 'userId': userId }};
228             self.serviceBusService.sendTopicMessage(self.topicName, message, function (err) {
229                 if (err) log.error(err);
230                 if (callback) callback(err);
231                 deferred.resolve();
232             });
233             return deferred.promise;
234         };
235 
236         self.postPromiseChain = self.postPromiseChain.then(proc, function (err) { log.error(err); });
237     },
238 
239     readLoop: function(monitor, callback) {
240         var self = this;
241         var log = self.log.child('readLoop ' + monitor.name);
242         log.BEGIN();
243         var proc = function () {
244             log.debug('calling receiveSubscriptionMessage');
245             self.serviceBusService.receiveSubscriptionMessage(self.topicName, monitor.name,
246                 function (err, found) {
247                     if (err && err != 'No messages to receive') {
248                         log.error(err);
249                         log.error('stopping read loop');
250                         return;
251                     }
252 
253                     if (found) {
254                         if (! callback(null, found)) {
255                             log.error('stopping read loop');
256                             return;
257                         }
258                     }
259 
260                     proc();
261                 });
262         };
263         proc();
264     },
265 
266     commandLoop: function() {
267         var self = this;
268         var log = self.log.child('commandLoop');
269         log.BEGIN();
270         self.readLoop(self.registrationMonitor, function(err, found) {
271             if (err) {
272 
273                 // readLoop stops when there is an error - restart ourselves.
274                 //
275                 self.commandLoop();
276             }
277             else if (found) {
278                 var command = found.body[0];
279                 var userId = found.body.slice(1);
280 
281                 // Monitoring activationg request - allow for a fixed interval
282                 //
283                 if (command == '+') {
284                     var expiration = new Date();
285                     expiration.setMinutes(expiration.getMinutes() + config.monitor_duration_minutes);
286                     log.debug('starting to monitor:', userId, 'expires:', expiration);
287                     self.userMonitoringExpirations[userId] = expiration;
288                     self.post(userId, '*** monitoring started ***');
289                 }
290 
291                 // Monitoring termination request. Note this wil terminate multiple monitors for the same user ID.
292                 //
293                 else if (command == '-') {
294                     log.debug('stop monitoring:', userId);
295                     delete self.userMonitoringExpirations[userId];
296                 }
297 
298                 else {
299                     log.error('invalid monitor channel command:', command);
300                 }
301             }
302 
303             // Keep the readLoop running.
304             //
305             return true;
306         });
307         log.END();
308     },
309 
310     cleanup: function() {
311         var self = this;
312         var log = self.log.child('cleanup');
313 
314         // Fetch the list of active subscriptions to our topic. Cull those that are older than twice the monitoring
315         // interval.
316         //
317         self.serviceBusService.listSubscriptions(self.topicName, function (err, subs, response) {
318             if (err) {
319                 log.error('failed listSubscriptions:', err);
320             }
321             else {
322                 log.debug('found', subs.length, 'subsciptions');
323                 var cutoff = new Date();
324                 cutoff.setHours(cutoff.getMinutes() - 2 * config.monitor_duration_minutes);
325                 for (var i = 0; i < subs.length; ++i) {
326                     var name = subs[i].SubscriptionName;
327                     var when = new Date(subs[i].AccessedAt);
328                     if (name == config.monitor_command_channel_name) {
329                         log.debug('skipping own command monitor');
330                         continue;
331                     }
332                     if (when < cutoff) {
333                         log.info('deleting stale subscription', name);
334                         self.serviceBusService.deleteSubscription(self.topicName, name,
335                             function (err, response) {
336                                 if (err) {
337                                     log.error('failed to delete subscription:', err);
338                                 }
339                                 else {
340                                     log.info('deleted');
341                                 }
342                             });
343                     }
344                     else {
345                         var remaining = when - cutoff;
346                         var hours = Math.floor(remaining / (60 * 60 * 1000));
347                         remaining -= hours * 60 * 60 * 1000;
348                         var mins = Math.floor(remaining / (60 * 1000));
349                         remaining -= mins * 60 * 1000;
350                         var secs = Math.floor(remaining / 1000);
351                         remaining = '';
352                         if (hours < 10) remaining += '0';
353                         remaining += hours + ':';
354                         if (mins < 10) remaining += '0';
355                         remaining += mins + ':';
356                         if (secs < 10) remaining += '0';
357                         remaining += secs;
358                         log.debug(name, 'will be removed in', remaining);
359                     }
360                 }
361             }
362 
363             setTimeout(self.cleanup.bind(self), config.monitor_cleanup_interval_minutes * 60 * 1000);
364         });
365     },
366 
367     createTopic: function(isServer, callback) {
368         var self = this;
369         var log = self.log.child('createTopic');
370         log.BEGIN('isServer:', isServer);
371         var proc = function() {
372             log.debug('calling createTopicIfNotExists');
373             self.serviceBusService.createTopicIfNotExists(self.topicName, self.topicConfig, function (err) {
374                 if (err) {
375                     log.error('failed createTopicIfNotExists:', err);
376                     if (err.statusCode === 409) { // Busy - try again after 1 second.
377                         log.debug('trying again in 1 second');
378                         Q.delay(1000).then(proc);
379                         return;
380                     }
381                     log.END(err);
382                     if (callback) callback(err);
383                 }
384                 else if (isServer) {
385 
386                     // Create the monitor for the command channel. This will listen for monitor start/stop commands.
387                     //
388                     log.debug('creating command channel monitor');
389                     self.makeMonitor(config.monitor_command_channel_name, function (err, monitor) {
390                         log.END(err);
391                         self.registrationMonitor = monitor;
392                         self.cleanup();
393                         self.commandLoop();
394                         if (callback) callback(err);
395                     });
396                 }
397                 else {
398                     if (callback) callback(null);
399                 }
400             });
401         };
402         proc();
403     }
